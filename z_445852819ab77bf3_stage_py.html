206 def evaluate_given_speed_and_target_discharge_pressure( 

207 self, 

208 inlet_stream_stage: FluidStream, 

209 mass_rate_kg_per_hour: float, 

210 target_discharge_pressure: float, 

211 speed: float | None = None, 

212 ) -> CompressorTrainStageResultSingleTimeStep: 

213 """ 

214 Calculate the result of a single-speed compressor stage given a target discharge pressure. 

215 

216 This method evaluates the compressor stage performance by iterating on the additional mass rate 

217 (ASV recirculation) to achieve the target discharge pressure. It ensures the solution is within 

218 the compressor's operational constraints. 

219 

220 The process involves: 

221 1. Evaluating the stage without recirculation to check if the target pressure is met. 

222 2. Evaluating the stage with maximum recirculation to check if the target pressure is exceeded. 

223 3. Using a root-finding algorithm to determine the additional mass rate required to meet the target 

224 discharge pressure if the solution lies between the two bounds. 

225 

226 Args: 

227 inlet_stream_stage (FluidStream): The inlet stream for the stage, containing pressure, temperature, 

228 and other fluid properties. 

229 mass_rate_kg_per_hour (float): The mass rate entering the stage in kilograms per hour [kg/hour]. 

230 target_discharge_pressure (float): The target discharge pressure for the stage in bar absolute [bara]. 

231 speed (float) 

232 

233 Returns: 

234 CompressorTrainStageResultSingleTimeStep: The result of the evaluation for the compressor stage, 

235 including the outlet stream and operational details. 

236 """ 

237 # If no speed is defined for VariableSpeedCompressorChart, use the minimum speed 

238 if isinstance(self.compressor_chart, VariableSpeedCompressorChart) and speed is None: 

239 speed = self.compressor_chart.minimum_speed 

240 

241 result_no_recirculation = self.evaluate( 

242 inlet_stream_stage=inlet_stream_stage, 

243 mass_rate_kg_per_hour=mass_rate_kg_per_hour, 

244 speed=speed, 

245 asv_additional_mass_rate=0, 

246 ) 

247 

248 # result_no_recirculation.inlet_stream.density_kg_per_m3 will have correct pressure and temperature 

249 # to find max mass rate, inlet_stream_stage will not 

250 maximum_rate = ( 

251 self.compressor_chart.maximum_rate 

252 if isinstance(self.compressor_chart, SingleSpeedCompressorChart) 

253 else self.compressor_chart.maximum_rate_as_function_of_speed(speed) 

254 ) 

255 

256 max_recirculation = max( 

257 maximum_rate * result_no_recirculation.inlet_stream.density_kg_per_m3 - mass_rate_kg_per_hour - EPSILON, 

258 0, 

259 ) 

260 result_max_recirculation = self.evaluate( 

261 inlet_stream_stage=inlet_stream_stage, 

262 mass_rate_kg_per_hour=mass_rate_kg_per_hour, 

263 asv_additional_mass_rate=max_recirculation, 

264 speed=speed, 

265 ) 

266 if result_no_recirculation.discharge_pressure < target_discharge_pressure: 

267 return result_no_recirculation 

268 elif result_max_recirculation.discharge_pressure > target_discharge_pressure: 

269 return result_max_recirculation

271 def _calculate_compressor_stage( 

272 additional_mass_rate: float, 

273 ) -> CompressorTrainStageResultSingleTimeStep: 

274 return self.evaluate( 

275 inlet_stream_stage=inlet_stream_stage, 

276 mass_rate_kg_per_hour=mass_rate_kg_per_hour, 

277 asv_additional_mass_rate=additional_mass_rate, 

278 speed=speed, 

279 ) 

280 

281 result_mass_rate = find_root( 

282 lower_bound=0, 

283 upper_bound=max_recirculation, 

284 func=lambda x: _calculate_compressor_stage(additional_mass_rate=x).discharge_pressure 

285 - target_discharge_pressure, 

286 ) 

287 

288 return _calculate_compressor_stage(result_mass_rate)
