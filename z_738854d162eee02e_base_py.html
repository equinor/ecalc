17 """This is used when merging different time slots when the energy function of a consumer changes over time. 

18 Append method covering all the basics. All additional extend methods needs to be covered in

22 other_values = other.__getattribute__(attribute)

25 logger.warning(

27 )

30 logger.warning(

33 ) 

34 elif isinstance(values, EnergyModelBaseResult): 

35 # In case of nested models such as compressor with turbine 

36 values.extend(other_values) 

37 elif isinstance(values, list):
