21class VariableSpeedCompressorTrainCommonShaftMultipleStreamsAndPressures( 

22 CompressorTrainModel[VariableSpeedCompressorTrainMultipleStreamsAndPressures] 

23): 

24 """An advanced model of a compressor train with variable speed, with the possibility of modelling additional 

25 streams going into or leaving the compressor train between the compressor train stages. 

26 

27 In general, a compressor train is a series of compressors that are running with the same shaft, meaning they will 

28 always have the same speed. Given inlet conditions (composition, temperature, pressure, rate) for all ingoing 

29 streams, pressures and rates for outgoing streams (not counting the outlet at the final stage - it is what it 

30 becomes), and a shaft speed, the intermediate pressures (and temperature before cooling) between stages and the 

31 outlet pressure (and temperature) is given. To solve this for a given outlet pressure, 

32 one must iterate to find the speed. 

33 

34 Streams: 

35 Models of the streams in the compressor train. See FluidStream 

36 Can be used with any number of streams going into or leaving the compressor train. 

37 The inlet of the first compressor stage must have an ingoing stream connected. 

38 

39 Stages: 

40 Models of the compressor train stages. See CompressorTrainStage. 

41 For each stage, one must specify a compressor chart, an inlet temperature and whether to take out liquids before 

42 compression. In addition, one must specify the pressure drop from previous stage (may be 0). 

43 

44 minimum_speed: 

45 The minimum speed of the compressor train shaft. 

46 If not specified it will be calculated from the given compressor charts for each of the stages 

47 

48 maximum_speed: 

49 The maximum speed of the compressor train shaft. 

50 If not specified it will be calculated from the given compressor charts for each of the stages 

51 

52 """

191 def check_that_ingoing_streams_are_larger_than_or_equal_to_outgoing_streams( 

192 self, 

193 std_rates_std_m3_per_day_per_stream: list[float], 

194 ) -> bool: 

195 """ 

196 Check that, for each stage in the compressor train, the cumulative sum of ingoing streams up to that stage 

197 is at least as large as the cumulative sum of outgoing streams. This ensures mass balance, as a compressor 

198 train cannot generate fluid. 

199 

200 Args: 

201 std_rates_std_m3_per_day_per_stream (list[float]): Standard rates [Sm3/day] for each stream in the compressor train. 

202 

203 Returns: 

204 bool: True if, at every stage, the total ingoing volume is greater than or equal to the outgoing volume; False otherwise. 

205 """ 

206 sum_of_ingoing_volume_up_to_current_stage_number = 0 

207 sum_of_outgoing_volume_up_to_current_stage_number = 0 

208 for stage_number, _ in enumerate(self.stages): 

209 sum_of_ingoing_volume_up_to_current_stage_number += sum( 

210 [ 

211 std_rates_std_m3_per_day_per_stream[inlet_stream_number] 

212 for inlet_stream_number in self.inlet_stream_connected_to_stage[stage_number] 

213 ] 

214 ) 

215 sum_of_outgoing_volume_up_to_current_stage_number += sum( 

216 [ 

217 std_rates_std_m3_per_day_per_stream[inlet_stream_number] 

218 for inlet_stream_number in self.outlet_stream_connected_to_stage[stage_number] 

219 ] 

220 ) 

221 if sum_of_outgoing_volume_up_to_current_stage_number > sum_of_ingoing_volume_up_to_current_stage_number: 

222 logger.warning( 

223 f"For stage number {stage_number}, the sum of the outgoing streams exceeds the sum of the ingoing " 

224 f"streams. Rates will be set to zero and time step not calculated." 

225 ) 

226 return False 

227 

228 return True

230 def convert_to_rates_for_each_compressor_train_stages(self, rates_per_stream: list[float]) -> list[float]: 

231 """ 

232 Convert stream rates to per-stage rates for the compressor train. 

233 

234 This method takes the rates for each stream (either mass rates [kg/h] or standard rates [Sm3/day]) 

235 and computes the corresponding rate for each stage in the compressor train, accounting for streams 

236 entering or leaving between stages. 

237 

238 Args: 

239 rates_per_stream (list[float]): Rates for each stream in the compressor train, either as mass rates [kg/h] 

240 or standard rates [Sm3/day]. 

241 

242 Returns: 

243 list[float]: Rates for each stage in the compressor train, in the same units as the input. 

244 """ 

245 stage_rates = [] 

246 current_compressor_stage = 0 

247 previous_compressor_stage = 0 

248 for i, stream in enumerate(self.streams): 

249 # the next stage with an additional ingoing/outgoing stream. 

250 # current_compressor_stage is not necessarily previous_compressor_stage + 1, unless streams are entering 

251 # or leaving the train between all stages 

252 current_compressor_stage = stream.connected_to_stage_no 

253 if i == 0: 

254 stage_rates.append(rates_per_stream[i]) 

255 else: 

256 # if no ingoing/outgoing streams between compressor train stages, 

257 # keep the rate coming out of the previous compressor train stage 

258 for _ in range(previous_compressor_stage, current_compressor_stage): 

259 stage_rates.append(stage_rates[previous_compressor_stage]) 

260 if stream.is_inlet_stream: # if there is an extra ingoing rate, then add it 

261 stage_rates[current_compressor_stage] += rates_per_stream[i] 

262 else: # if there is an extra outgoing rate, then add it 

263 stage_rates[current_compressor_stage] -= rates_per_stream[i] 

264 previous_compressor_stage = current_compressor_stage 

265 

266 # if there are more stages after the last ingoing/outgoing stream keep the same rate 

267 for _ in range(current_compressor_stage + 1, len(self.stages)): 

268 stage_rates.append(stage_rates[previous_compressor_stage]) 

269 

270 return stage_rates

358 def calculate_compressor_train( 

359 self, 

360 constraints: CompressorTrainEvaluationInput, 

361 asv_rate_fraction: float = 0.0, 

362 asv_additional_mass_rate: float = 0.0, 

363 ) -> CompressorTrainResultSingleTimeStep: 

364 """ 

365 Simulate the compressor train for the given inlet conditions, stream rates, and shaft speed. 

366 

367 This method models the flow through each stage of the compressor train, accounting for multiple inlet and outlet 

368 streams, anti-surge valve recirculation, and possible subtrain configurations. It computes the resulting outlet 

369 stream, per-stage results (including conditions and power), and overall train performance. 

370 

371 Typical usage scenarios: 

372 1. Standard train: self.streams[0] is the main inlet. 

373 2. First subtrain (with intermediate pressure target): self.streams[0] is the inlet. 

374 3. Last subtrain (after a split): self.streams[0] may be an entering or leaving stream, and the inlet fluid 

375 is set from the first subtrain. 

376 

377 Args: 

378 constraints (CompressorTrainEvaluationInput): Pressures, stream rates, and speed for the simulation. 

379 asv_rate_fraction (float, optional): Fraction of anti-surge valve recirculation. Defaults to 0.0. 

380 asv_additional_mass_rate (float, optional): Additional mass rate for recirculation. Defaults to 0.0. 

381 

382 Returns: 

383 CompressorTrainResultSingleTimeStep: Object containing inlet/outlet streams, per-stage results, speed, 

384 total power, and pressure status. 

385 """ 

386 # This method requires suction_pressure and rate to be set 

387 assert constraints.suction_pressure is not None 

388 assert constraints.rate is not None 

389 # This multiple streams train also requires stream_rates to be set 

390 assert constraints.stream_rates is not None 

391 

392 stage_results = [] 

393 train_inlet_stream = inlet_stream = self.streams[0].fluid.get_fluid_stream( 

394 pressure_bara=constraints.suction_pressure, 

395 temperature_kelvin=self.stages[0].inlet_temperature_kelvin, 

396 ) 

397 mass_rate_this_stage_kg_per_hour = inlet_stream.standard_rate_to_mass_rate(constraints.rate) 

398 mass_rate_previous_stage_kg_per_hour = 0 

399 previous_outlet_stream = None 

400 

401 for stage_number, stage in enumerate(self.stages): 

402 if stage_number > 0: 

403 inlet_stream = previous_outlet_stream 

404 mass_rate_this_stage_kg_per_hour = mass_rate_previous_stage_kg_per_hour 

405 # take mass out before potential mixing with additional ingoing stream 

406 # assume that volume/mass is always taken out before additional volume/mass is potentially added, no matter 

407 # what order the streams are defined in the yaml file 

408 for stream_number in self.outlet_stream_connected_to_stage.get(stage_number): 

409 mass_rate_this_stage_kg_per_hour -= inlet_stream.standard_rate_to_mass_rate( 

410 constraints.stream_rates[stream_number] 

411 ) 

412 for stream_number in self.inlet_stream_connected_to_stage.get(stage_number): 

413 if stream_number > 0: 

414 # Flash at outlet temperature or inlet temperature next stage? Must assume same temperature. 

415 # Currently, using outlet temperature 

416 additional_inlet_stream = self.streams[stream_number].fluid.get_fluid_stream( 

417 pressure_bara=inlet_stream.pressure_bara, 

418 temperature_kelvin=inlet_stream.temperature_kelvin, 

419 ) 

420 mass_rate_additional_inlet_stream_kg_per_hour = additional_inlet_stream.standard_rate_to_mass_rate( 

421 float(constraints.stream_rates[stream_number]) 

422 ) 

423 if (mass_rate_this_stage_kg_per_hour > 0) or (mass_rate_additional_inlet_stream_kg_per_hour > 0): 

424 inlet_stream = additional_inlet_stream.mix_in_stream( 

425 other_fluid_stream=inlet_stream, 

426 self_mass_rate=mass_rate_additional_inlet_stream_kg_per_hour, # type: ignore[arg-type] 

427 other_mass_rate=mass_rate_this_stage_kg_per_hour, # type: ignore[arg-type] 

428 temperature_kelvin=additional_inlet_stream.temperature_kelvin, 

429 pressure_bara=additional_inlet_stream.pressure_bara, 

430 ) 

431 mass_rate_this_stage_kg_per_hour += mass_rate_additional_inlet_stream_kg_per_hour 

432 

433 if mass_rate_this_stage_kg_per_hour == 0: 

434 fluid_to_recirculate = self.get_fluid_to_recirculate_in_stage_when_inlet_rate_is_zero( 

435 stage_number=stage_number 

436 ) 

437 if fluid_to_recirculate: 

438 logger.warning( 

439 f"For stage number {stage_number}, there is no fluid entering the stage at at this time step. " 

440 f"The compressor is only recirculating fluid. Standard rates are " 

441 f"{constraints.stream_rates}." 

442 ) 

443 inlet_stream = fluid_to_recirculate 

444 else: 

445 raise ValueError( 

446 f"Trying to recirculate fluid in stage {stage_number} without defining which " 

447 f"composition the fluid should have." 

448 ) 

449 

450 stage_result = stage.evaluate( 

451 inlet_stream_stage=inlet_stream, 

452 mass_rate_kg_per_hour=mass_rate_this_stage_kg_per_hour, # type: ignore[arg-type] 

453 speed=constraints.speed, 

454 asv_rate_fraction=asv_rate_fraction, 

455 asv_additional_mass_rate=asv_additional_mass_rate, 

456 ) 

457 stage_results.append(stage_result) 

458 

459 # We need to recreate the domain object from the result object. This needs cleaning up. 

460 previous_outlet_stream = inlet_stream.set_new_pressure_and_temperature( 

461 new_pressure_bara=stage_result.outlet_stream.pressure_bara, 

462 new_temperature_kelvin=stage_result.outlet_stream.temperature_kelvin, 

463 ) 

464 

465 mass_rate_previous_stage_kg_per_hour = mass_rate_this_stage_kg_per_hour 

466 self.set_fluid_to_recirculate_in_stage_when_inlet_rate_is_zero( 

467 stage_number=stage_number, fluid_stream=inlet_stream 

468 ) 

469 

470 # check if target pressures are met 

471 target_pressure_status = self.check_target_pressures( 

472 constraints=constraints, 

473 results=stage_results, 

474 ) 

475 

476 return CompressorTrainResultSingleTimeStep( 

477 inlet_stream=FluidStreamDTO.from_fluid_domain_object(fluid_stream=train_inlet_stream), 

478 outlet_stream=FluidStreamDTO.from_fluid_domain_object(fluid_stream=previous_outlet_stream), 

479 stage_results=stage_results, 

480 speed=constraints.speed or float("nan"), 

481 above_maximum_power=sum([stage_result.power_megawatt for stage_result in stage_results]) 

482 > self.maximum_power 

483 if self.maximum_power 

484 else False, 

485 target_pressure_status=target_pressure_status, 

486 )

488 def _update_inlet_fluid_and_std_rates_for_last_subtrain( 

489 self, 

490 std_rates_std_m3_per_day_per_stream: list[float], 

491 inlet_pressure: float, 

492 ) -> tuple[FluidStream, list[float]]: 

493 """ 

494 Updates the inlet fluid and standard rates for the last subtrain after splitting a compressor train. 

495 

496 This method constructs the inlet fluid stream for the last subtrain by mixing the appropriate streams 

497 based on the provided standard rates and inlet pressure. It also computes the updated list of standard 

498 rates for the subtrain, ensuring that outgoing and additional ingoing streams are handled correctly. 

499 

500 Args: 

501 std_rates_std_m3_per_day_per_stream (list[float]): Standard rates for each stream in the compressor train [Sm3/day]. 

502 inlet_pressure (float): The inlet pressure for the subtrain [bara]. 

503 

504 Returns: 

505 tuple[FluidStream, list[float]]: A tuple containing: 

506 - The updated inlet fluid stream for the subtrain. 

507 - The updated list of standard rates for the subtrain. 

508 """ 

509 # first make inlet stream from stream[0] 

510 inlet_stream = self.streams[0].fluid.get_fluid_stream( 

511 pressure_bara=inlet_pressure, 

512 temperature_kelvin=self.stages[0].inlet_temperature_kelvin, 

513 ) 

514 inlet_std_rate = std_rates_std_m3_per_day_per_stream[0] 

515 inlet_mass_rate = inlet_stream.standard_rate_to_mass_rate(inlet_std_rate) 

516 

517 # take mass out before potential mixing with additional ingoing stream 

518 # assume that volume/mass is always taken out before additional volume/mass is potentially added, no matter 

519 # what order the streams are defined in the yaml file 

520 for stream_number in self.outlet_stream_connected_to_stage.get(0): 

521 inlet_std_rate -= std_rates_std_m3_per_day_per_stream[stream_number] 

522 for stream_number in self.inlet_stream_connected_to_stage.get(0): 

523 if stream_number > 0: 

524 # mix stream from previous subtrain with incoming stream at first stage 

525 additional_inlet_stream = self.streams[stream_number].fluid.get_fluid_stream( 

526 pressure_bara=inlet_pressure, 

527 temperature_kelvin=self.stages[0].inlet_temperature_kelvin, 

528 ) 

529 mass_rate_additional_inlet_stream = additional_inlet_stream.standard_rate_to_mass_rate( 

530 std_rates_std_m3_per_day_per_stream[stream_number] 

531 ) 

532 # mix streams to get inlet stream for first compressor stage 

533 # if rate is 0 don't try to mix, 

534 if (inlet_mass_rate > 0) or (mass_rate_additional_inlet_stream > 0): 

535 inlet_stream = additional_inlet_stream.mix_in_stream( 

536 other_fluid_stream=inlet_stream, 

537 self_mass_rate=mass_rate_additional_inlet_stream, # type: ignore[arg-type] 

538 other_mass_rate=inlet_mass_rate, # type: ignore[arg-type] 

539 temperature_kelvin=additional_inlet_stream.temperature_kelvin, 

540 pressure_bara=additional_inlet_stream.pressure_bara, 

541 ) 

542 inlet_std_rate += std_rates_std_m3_per_day_per_stream[stream_number] 

543 inlet_mass_rate = inlet_stream.standard_rate_to_mass_rate(inlet_std_rate) 

544 

545 # update inlet fluid for the subtrain with new composition 

546 if inlet_mass_rate == 0: 

547 fluid_to_recirculate = self.get_fluid_to_recirculate_in_stage_when_inlet_rate_is_zero(stage_number=0) 

548 if fluid_to_recirculate: 

549 updated_inlet_fluid = fluid_to_recirculate 

550 else: 

551 raise ValueError("Trying to recirculate unknown fluid in compressor stage") 

552 else: 

553 updated_inlet_fluid = FluidStream(fluid_model=inlet_stream.fluid_model) 

554 updated_std_rates_std_m3_per_day_per_stream = [inlet_std_rate] 

555 updated_std_rates_std_m3_per_day_per_stream.extend( 

556 [ 

557 std_rates_std_m3_per_day_per_stream[stream_number] 

558 for stream_number in range(self.number_of_compressor_streams) 

559 if stream_number 

560 not in set(self.outlet_stream_connected_to_stage.get(0) + self.inlet_stream_connected_to_stage.get(0)) # type: ignore[operator] 

561 ] 

562 ) 

563 

564 return updated_inlet_fluid, updated_std_rates_std_m3_per_day_per_stream

566 def find_and_calculate_for_compressor_train_with_two_pressure_requirements( 

567 self, 

568 stage_number_for_intermediate_pressure_target: int, 

569 constraints: CompressorTrainEvaluationInput, 

570 pressure_control_first_part: FixedSpeedPressureControl, 

571 pressure_control_last_part: FixedSpeedPressureControl, 

572 ) -> CompressorTrainResultSingleTimeStep: 

573 """ 

574 Calculates the compressor train performance when two pressure targets are specified: an intermediate (interstage) 

575 pressure and a final discharge pressure. 

576 

577 The method splits the compressor train into two sub-trains at the specified stage. It determines the optimal shaft 

578 speed for each sub-train to meet their respective pressure targets, applies the appropriate pressure control 

579 strategies, and combines the results. 

580 

581 Typical use case: part of the gas is taken out at an intermediate stage (e.g., for export or injection), 

582 requiring both an interstage and a final discharge pressure to be met. 

583 

584 Args: 

585 stage_number_for_intermediate_pressure_target (int): The stage index at which the intermediate pressure target applies. 

586 constraints (CompressorTrainEvaluationInput): The operating constraints, including pressures and stream rates. 

587 pressure_control_first_part (FixedSpeedPressureControl): Pressure control strategy for the first sub-train. 

588 pressure_control_last_part (FixedSpeedPressureControl): Pressure control strategy for the second sub-train. 

589 

590 Returns: 

591 CompressorTrainResultSingleTimeStep: The combined result of the two sub-trains, including stage results, 

592 inlet/outlet streams, speed, and pressure status. 

593 """ 

594 # This method requires stream_rates to be set for splitting operations 

595 assert constraints.stream_rates is not None 

596 

597 # Split train into two and calculate minimum speed to reach required pressures 

598 compressor_train_first_part, compressor_train_last_part = split_train_on_stage_number( 

599 compressor_train=self, 

600 stage_number=stage_number_for_intermediate_pressure_target, 

601 pressure_control_first_part=pressure_control_first_part, 

602 pressure_control_last_part=pressure_control_last_part, 

603 ) 

604 

605 std_rates_first_part, std_rates_last_part = split_rates_on_stage_number( 

606 compressor_train=self, 

607 rates_per_stream=constraints.stream_rates, 

608 stage_number=stage_number_for_intermediate_pressure_target, 

609 ) 

610 constraints_first_part = CompressorTrainEvaluationInput( 

611 suction_pressure=constraints.suction_pressure, 

612 discharge_pressure=constraints.interstage_pressure, 

613 rate=std_rates_first_part[0], 

614 stream_rates=std_rates_first_part, 

615 ) 

616 constraints_last_part = CompressorTrainEvaluationInput( 

617 suction_pressure=constraints.interstage_pressure, 

618 discharge_pressure=constraints.discharge_pressure, 

619 rate=std_rates_last_part[0], 

620 stream_rates=std_rates_last_part, 

621 ) 

622 

623 compressor_train_first_part_optimal_speed = compressor_train_first_part.find_shaft_speed_given_constraints( 

624 constraints=constraints_first_part, 

625 lower_bound_for_speed=self.minimum_speed, # Only search for a solution within the bounds of the 

626 upper_bound_for_speed=self.maximum_speed, # original, complete compressor train 

627 ) 

628 compressor_train_results_first_part_with_optimal_speed_result = ( 

629 compressor_train_first_part.calculate_compressor_train( 

630 constraints=constraints_first_part.create_conditions_with_new_input( 

631 new_speed=compressor_train_first_part_optimal_speed, 

632 ) 

633 ) 

634 ) 

635 

636 if self.data_transfer_object.calculate_max_rate: 

637 max_standard_rate_per_stream = [ 

638 compressor_train_first_part.get_max_standard_rate( # type: ignore[call-arg] 

639 constraints=CompressorTrainEvaluationInput( 

640 rate=std_rates_first_part[0], 

641 suction_pressure=constraints_first_part.suction_pressure, 

642 discharge_pressure=constraints_first_part.discharge_pressure, 

643 stream_rates=std_rates_first_part, 

644 stream_to_maximize=stream_index, 

645 ), 

646 ) 

647 for stream_index, _ in enumerate(compressor_train_first_part.streams) 

648 ] 

649 else: 

650 max_standard_rate_per_stream = [float("nan")] * len(std_rates_first_part) # type: ignore[list-item] 

651 

652 # set self.inlet_fluid based on outlet_stream_first_part 

653 compressor_train_last_part.streams[0].fluid = FluidStream( # filling the placeholder with the correct fluid 

654 fluid_model=FluidModel( 

655 composition=compressor_train_results_first_part_with_optimal_speed_result.stage_results[ 

656 -1 

657 ].outlet_stream.composition, 

658 eos_model=compressor_train_results_first_part_with_optimal_speed_result.stage_results[ 

659 -1 

660 ].outlet_stream.eos_model, 

661 ) 

662 ) 

663 

664 compressor_train_last_part_optimal_speed = compressor_train_last_part.find_shaft_speed_given_constraints( 

665 constraints=constraints_last_part, 

666 lower_bound_for_speed=self.minimum_speed, 

667 upper_bound_for_speed=self.maximum_speed, 

668 ) 

669 compressor_train_results_last_part_with_optimal_speed_result = ( 

670 compressor_train_last_part.calculate_compressor_train( 

671 constraints=constraints_last_part.create_conditions_with_new_input( 

672 new_speed=compressor_train_last_part_optimal_speed, 

673 ) 

674 ) 

675 ) 

676 

677 if self.data_transfer_object.calculate_max_rate: 

678 for stream_index, _ in enumerate(compressor_train_last_part.streams): 

679 if stream_index > 0: 

680 max_standard_rate_per_stream.append( 

681 compressor_train_first_part.get_max_standard_rate( # type: ignore[call-arg] 

682 constraints=CompressorTrainEvaluationInput( 

683 rate=std_rates_last_part[0], 

684 suction_pressure=constraints_last_part.suction_pressure, 

685 discharge_pressure=constraints_last_part.discharge_pressure, 

686 stream_rates=std_rates_last_part, 

687 stream_to_maximize=stream_index, 

688 ), 

689 ) 

690 ) 

691 else: 

692 max_standard_rate_per_stream = max_standard_rate_per_stream + [float("nan")] * len(std_rates_last_part[1:]) # type: ignore[list-item] 

693 """ 

694 Determine which of the first and last part will govern the speed to use 

695 Then run the last part as a single speed train with the speed chosen 

696 Fixme: Need to deliver the result in a proper format below. 

697 """ 

698 if compressor_train_first_part_optimal_speed > compressor_train_last_part_optimal_speed: 

699 speed = compressor_train_first_part_optimal_speed 

700 compressor_train_results_last_part_with_pressure_control = ( 

701 compressor_train_last_part.evaluate_with_pressure_control_given_constraints( 

702 constraints=constraints_last_part.create_conditions_with_new_input( 

703 new_speed=speed, 

704 ) 

705 ) 

706 ) 

707 compressor_train_results_to_return_first_part = ( 

708 compressor_train_results_first_part_with_optimal_speed_result 

709 ) 

710 compressor_train_results_to_return_last_part = compressor_train_results_last_part_with_pressure_control 

711 

712 else: 

713 speed = compressor_train_last_part_optimal_speed 

714 compressor_train_results_first_part_with_pressure_control = ( 

715 compressor_train_first_part.evaluate_with_pressure_control_given_constraints( 

716 constraints=constraints_first_part.create_conditions_with_new_input( 

717 new_speed=speed, 

718 ) 

719 ) 

720 ) 

721 compressor_train_results_to_return_first_part = compressor_train_results_first_part_with_pressure_control 

722 compressor_train_results_to_return_last_part = compressor_train_results_last_part_with_optimal_speed_result 

723 

724 compressor_train_results_to_return_stage_results = list( 

725 compressor_train_results_to_return_first_part.stage_results 

726 + compressor_train_results_to_return_last_part.stage_results 

727 ) 

728 

729 for stage_number in range(len(self.stages)): 

730 self.set_fluid_to_recirculate_in_stage_when_inlet_rate_is_zero( 

731 stage_number=stage_number, 

732 fluid_stream=compressor_train_first_part.get_fluid_to_recirculate_in_stage_when_inlet_rate_is_zero( 

733 stage_number=stage_number 

734 ) 

735 if stage_number < stage_number_for_intermediate_pressure_target 

736 else compressor_train_last_part.get_fluid_to_recirculate_in_stage_when_inlet_rate_is_zero( 

737 stage_number=stage_number - stage_number_for_intermediate_pressure_target 

738 ), 

739 ) 

740 

741 train_result = CompressorTrainResultSingleTimeStep( 

742 inlet_stream=compressor_train_results_to_return_first_part.inlet_stream, 

743 outlet_stream=compressor_train_results_to_return_last_part.outlet_stream, 

744 speed=speed, 

745 stage_results=compressor_train_results_to_return_stage_results, 

746 above_maximum_power=sum( 

747 [stage_result.power_megawatt for stage_result in compressor_train_results_to_return_stage_results] 

748 ) 

749 > self.maximum_power 

750 if self.maximum_power 

751 else False, 

752 target_pressure_status=TargetPressureStatus.NOT_CALCULATED, 

753 ) 

754 

755 # check if target pressures are met 

756 target_pressure_status = self.check_target_pressures( 

757 constraints=constraints, 

758 results=train_result, 

759 ) 

760 

761 train_result.target_pressure_status = target_pressure_status 

762 

763 return train_result

797def split_rates_on_stage_number( 

798 compressor_train: VariableSpeedCompressorTrainCommonShaftMultipleStreamsAndPressures, 

799 rates_per_stream: list[float], 

800 stage_number: int, 

801) -> tuple[list[float], list[float]]: 

802 """ 

803 Splits the stream rates for a compressor train into two lists at the specified stage number. 

804 

805 The first list contains rates for streams connected to stages before the split. 

806 The second list contains the rate at the split stage and rates for streams connected to stages at or after the split. 

807 

808 Args: 

809 compressor_train: The compressor train instance containing stream and stage information. 

810 rates_per_stream: List of rates for each stream in the compressor train. 

811 stage_number: The stage index at which to split the rates (0-based). 

812 

813 Returns: 

814 Tuple containing: 

815 - List of rates for the first part (before the split stage). 

816 - List of rates for the last part (from the split stage onward). 

817 """ 

818 rates_first_part = [ 

819 rates_per_stream[i] 

820 for i, stream in enumerate(compressor_train.streams) 

821 if stream.connected_to_stage_no < stage_number 

822 ] 

823 

824 rates_last_part = [ 

825 compressor_train.convert_to_rates_for_each_compressor_train_stages(rates_per_stream)[stage_number - 1] 

826 ] 

827 

828 for i, stream in enumerate(compressor_train.streams): 

829 if stream.connected_to_stage_no >= stage_number: 

830 rates_last_part.append(rates_per_stream[i]) 

831 

832 return rates_first_part, rates_last_part

835def split_train_on_stage_number( 

836 compressor_train: VariableSpeedCompressorTrainCommonShaftMultipleStreamsAndPressures, 

837 stage_number: int, 

838 pressure_control_first_part: FixedSpeedPressureControl | None = None, 

839 pressure_control_last_part: FixedSpeedPressureControl | None = None, 

840) -> tuple[ 

841 VariableSpeedCompressorTrainCommonShaftMultipleStreamsAndPressures, 

842 VariableSpeedCompressorTrainCommonShaftMultipleStreamsAndPressures, 

843]: 

844 """ 

845 Splits a variable speed compressor train into two sub-trains at the specified stage number. 

846 

847 The first sub-train includes all stages before the split, and the second sub-train includes all stages from the split onward. 

848 Optionally, different pressure control strategies can be assigned to each sub-train. 

849 

850 Args: 

851 compressor_train: The original variable speed compressor train to split. 

852 stage_number: The stage index at which to split the train (0-based). 

853 pressure_control_first_part: Optional pressure control for the first sub-train. 

854 pressure_control_last_part: Optional pressure control for the second sub-train. 

855 

856 Returns: 

857 Tuple containing: 

858 - The first sub-train (stages before the split). 

859 - The second sub-train (stages from the split onward). 

860 """ 

861 first_part_data_transfer_object = deepcopy(compressor_train.data_transfer_object) 

862 last_part_data_transfer_object = deepcopy(compressor_train.data_transfer_object) 

863 first_part_data_transfer_object.stages = first_part_data_transfer_object.stages[:stage_number] 

864 last_part_data_transfer_object.stages = last_part_data_transfer_object.stages[stage_number:] 

865 first_part_data_transfer_object.pressure_control = pressure_control_first_part 

866 last_part_data_transfer_object.pressure_control = pressure_control_last_part 

867 

868 compressor_train_first_part = VariableSpeedCompressorTrainCommonShaftMultipleStreamsAndPressures( 

869 streams=[stream for stream in compressor_train.streams if stream.connected_to_stage_no < stage_number], 

870 data_transfer_object=first_part_data_transfer_object, 

871 ) 

872 

873 streams_last_part = [ 

874 deepcopy(compressor_train.streams[compressor_train.inlet_stream_connected_to_stage.get(0)[0]]) # type: ignore[index] 

875 ] # placeholder for stream coming out of first train, updated at runtime 

876 streams_last_part.extend( 

877 [ 

878 FluidStreamObjectForMultipleStreams( 

879 fluid=stream.fluid, 

880 is_inlet_stream=stream.is_inlet_stream, 

881 connected_to_stage_no=stream.connected_to_stage_no - stage_number, 

882 ) 

883 for stream in compressor_train.streams 

884 if stream.connected_to_stage_no >= stage_number 

885 ] 

886 ) 

887 

888 compressor_train_last_part = VariableSpeedCompressorTrainCommonShaftMultipleStreamsAndPressures( 

889 streams=streams_last_part, 

890 data_transfer_object=last_part_data_transfer_object, 

891 ) 

892 

893 for stage_no in range(len(compressor_train.stages)): 

894 if stage_no < stage_number: 

895 compressor_train_first_part.set_fluid_to_recirculate_in_stage_when_inlet_rate_is_zero( 

896 stage_number=stage_no, 

897 fluid_stream=compressor_train.get_fluid_to_recirculate_in_stage_when_inlet_rate_is_zero(stage_no), 

898 ) 

899 else: 

900 compressor_train_last_part.set_fluid_to_recirculate_in_stage_when_inlet_rate_is_zero( 

901 stage_number=stage_no - stage_number, 

902 fluid_stream=compressor_train.get_fluid_to_recirculate_in_stage_when_inlet_rate_is_zero(stage_no), 

903 ) 

904 

905 return compressor_train_first_part, compressor_train_last_part
