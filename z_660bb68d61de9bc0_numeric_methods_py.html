13def find_root( 

14 lower_bound: float, 

15 upper_bound: float, 

16 func: Callable, 

17 relative_convergence_tolerance: float = CONVERGENCE_TOLERANCE, 

18 maximum_number_of_iterations: int = MAXIMUM_NUMBER_OF_ITERATIONS, 

19) -> float: 

20 """Root finding using scipy´s implementation of the brenth method. 

21 

22 TODO: Investigate why we don't use brentq method recommended by scipy 

23 

24 This will try to solve for the root: f(x) = 0. Another way to say this is "what x makes the function return 0"... 

25 

26 The result is bound on the interval [x0, x1]. 

27 

28 brenth is a version of Brent´s method (https://en.wikipedia.org/wiki/Brent%27s_method) with hyperbolic extrapolation 

29 

30 :param lower_bound: Lower bound of solution. Used as initial guess 

31 :param upper_bound: Upper bound of solution. Used as second guess 

32 :param func: The function to be used in the secant root-finding method that we will solve f(x) = 0 

33 :param relative_convergence_tolerance: The tolerance of convergence that will be used to exist the iteration 

34 :param maximum_number_of_iterations: The maximum number of iterations that will be used to find the root. 

35 """ 

36 try: 

37 result = root_scalar( 

38 func, 

39 bracket=(lower_bound, upper_bound), 

40 x0=lower_bound, 

41 x1=upper_bound, 

42 maxiter=maximum_number_of_iterations, 

43 method="brenth", 

44 rtol=relative_convergence_tolerance, 

45 ) 

46 if not result.converged: 

47 msg = ( 

48 f"Did not reach convergence after maximum number of iterations: {maximum_number_of_iterations}." 

49 f" root given last iteration: {result.root}, convergence_tolerance: {relative_convergence_tolerance}." 

50 f" on interval [{lower_bound},{upper_bound}]." 

51 f" func: {func}" 

52 ) 

53 logger.error(msg) 

54 return result.root 

55 except Exception as e: 

56 logger.error( 

57 f"Failed to find roots using Brent's method. Please report to the eCalc team." 

58 f" Bound to the interval [{lower_bound}, {upper_bound}], function: {func}" 

59 f" Max iterations: {maximum_number_of_iterations}, convergence_tolerance: {relative_convergence_tolerance}." 

60 f" Fallback to secant method bounded on the interval [{lower_bound}, {upper_bound}]. " + str(e) 

61 ) 

62 return secant_method( 

63 x0=lower_bound, 

64 x1=upper_bound, 

65 func=func, 

66 convergence_tolerance=relative_convergence_tolerance, 

67 maximum_number_of_iterations=maximum_number_of_iterations, 

68 )

71def secant_method( 

72 x0: float, 

73 x1: float, 

74 func: Callable, 

75 convergence_tolerance: float = CONVERGENCE_TOLERANCE, 

76 maximum_number_of_iterations: int = MAXIMUM_NUMBER_OF_ITERATIONS, 

77 bounded_to_input_x_interval: bool = True, 

78) -> float: 

79 """Keeping for now as fallback for root finding method above. 

80 

81 Root finding using the Secant method. 

82 

83 since we want to find the x that solves for the root: f(x) = 0. bound to the interval [x0, x1] 

84 Ref. https://en.wikipedia.org/wiki/Secant_method. 

85 :param x0: Starting-point for iteration together with x1 

86 :param x1: Starting-point for iteration together with x0 

87 :param func: The function to be used in the secant root-finding method that we will solve f(x) = 0 

88 :param convergence_tolerance: The tolerance of convergence that will be used to exist the iteration 

89 :param maximum_number_of_iterations: The maximum number of iterations that will be used to find the root. 

90 :param bounded_to_input_x_interval: Whether to search outside the input interval [x0, x1] for roots. 

91 """ 

92 f = func 

93 minimum_x_value = min(x0, x1) 

94 maximum_x_value = max(x0, x1) 

95 

96 def limit_to_bounds(x: Union[int, float], x_min: Union[int, float], x_max: Union[int, float]) -> Union[int, float]: 

97 return max(min(x, x_max), x_min) if bounded_to_input_x_interval else x 

98 

99 x2 = np.nan # Dummy value before iteration 

100 i = 0 

101 f0 = 0 

102 f1 = 1 

103 

104 rel_diff = 100.0 

105 # Loop until we reach the target tolerance or max iterations. 

106 while (abs(rel_diff) > convergence_tolerance) and (i <= maximum_number_of_iterations) and (f1 != f0): 

107 f0 = f(x0) 

108 f1 = f(x1) 

109 x2 = x1 - f1 * (x1 - x0) / float(f1 - f0) 

110 x2 = limit_to_bounds(x2, minimum_x_value, maximum_x_value) 

111 x0, x1 = x1, x2 # Update indexes for next iteration 

112 i += 1 

113 # Avoid division by zero: https://en.wikipedia.org/wiki/Relative_change_and_difference 

114 rel_diff = 0 if x0 == x1 else abs(x1 - x0) / max(abs(x0), abs(x1)) 

115 

116 if i > maximum_number_of_iterations: 

117 msg = ( 

118 f"Did not reach convergence after maximum number of iterations: {maximum_number_of_iterations}." 

119 f" x0: {x0}, x1: {x1}, convergence_tolerance: {convergence_tolerance}." 

120 f" bounded to input interval: {bounded_to_input_x_interval}" 

121 f" on interval [{minimum_x_value},{maximum_x_value}]." 

122 f" func: {func}" 

123 ) 

124 logger.error(msg) 

125 return x2

128def maximize_x_given_boolean_condition_function( 

129 x_min: float, 

130 x_max: float, 

131 bool_func: Callable, 

132 convergence_tolerance: float = CONVERGENCE_TOLERANCE, 

133 maximum_number_of_iterations: int = MAXIMUM_NUMBER_OF_ITERATIONS, 

134) -> float: 

135 """Binary search until we reach the maximum x value constrained by x_min and x_max 

136 where we have a boolean constraint condition given as a function. 

137 

138 max(x) given f(x) == True 

139 

140 We assume f(x) to be a binary (Heaviside step) function where f(x) is 1 for x <= n and 0 for x > n. 

141 n is the target value in this optimization. x == n is the highest possible value of x before f(x) turns to 0. 

142 

143 Note: This requires that the boolean condition is an indicator function where x > threshold returns False. 

144 """ 

145 x0, x1 = x_min, x_max 

146 x2 = (x0 + x1) / 2 # Initial value x2. 

147 i = 0 

148 rel_diff = 100.0 

149 

150 while (abs(rel_diff) > convergence_tolerance) and (i <= maximum_number_of_iterations): 

151 x2 = (x0 + x1) / 2 # Bisecting x0 and x1. 

152 if bool_func(x2): 

153 x0, x1 = x2, x1 # x2 is valid. We can now search to the right in the binary three. 

154 # Avoid division by zero: https://en.wikipedia.org/wiki/Relative_change_and_difference 

155 rel_diff = 0 if x0 == x1 else abs(x1 - x0) / max(abs(x0), abs(x1)) 

156 else: 

157 x0, x1 = x0, x2 # x2 is invalid. We can now search to the left in the binary three 

158 # rel_diff does not change since the solution is not valid. Otherwise we could accept a non-True solution. 

159 i += 1 

160 

161 if i > maximum_number_of_iterations: 

162 msg = ( 

163 f"Did not reach convergence after maximum number of iterations: {maximum_number_of_iterations}." 

164 f" x_min: {x_min}, x_max: {x_max}, convergence_tolerance: {convergence_tolerance}." 

165 f" bool_func: {bool_func}" 

166 ) 

167 logger.error(msg) 

168 return x2
