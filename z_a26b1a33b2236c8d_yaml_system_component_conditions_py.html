26 "Example 1:\n" 

27 "Two consumers where there is a cross-over such that if the rate for the first consumer exceeds its capacity,"

44 def ensure_one_crossover_out(cls, crossover: Optional[list[YamlCrossover]]): 

45 if crossover is None: 

46 return None 

47 crossover_out = [c.from_ for c in crossover]
