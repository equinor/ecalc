33class CompressorTrainSimplified(CompressorTrainModel): 

34 """A simplified model of a compressor train. 

35 

36 In general, a compressor train (series of compressors) are running with the same shaft, meaning they will always 

37 have the same speed. Given inlet fluid conditions (composition, temperature, pressure, rate) and a shaft speed, the 

38 intermediate pressures (and temperature before cooling) between stages and the outlet pressure (and temperature) is 

39 given. To solve this for a given outlet pressure, one must iterate to find the speed. 

40 

41 Simplification: 

42 The simplified approach used here, will instead make an assumption on the pressure ratio per stage and neglect 

43 the common shaft speed. Given an inlet and an outlet pressure for the train, the pressure ration for each stage is 

44 assumed equal and with this the inter-stage pressures are calculated. From this, the theoretical work is calculated 

45 for each compressor independently, neglecting the common shaft dependency. 

46 

47 float of compressors: 

48 There are in essence two ways of determining the number of compressors in the train: 

49 - Either with a compressor chart generator model with a predefined number of charts - one for each compressor. 

50 - Or, when using a compressor chart generator model where charts are estimated on input data together with 

51 setting compressor_maximum_pressure_ratio. In this case, the maximum total pressure ratio is found from the input 

52 data and the number of compressors are set such that this maximum is not exceeded for any of the stages. 

53 

54 Compressor chart (generator): 

55 For the simplified models, generic charts which are calculated at run-time is used, thus we do not have the chart 

56 up front, but need to calculate the chart at run-time, thus it uses chart-generators instead of charts directly. 

57 

58 There are three options to specify the compressor chart for each compressor in the train, two of these based on a 

59 generic unified compressor chart which is scaled given a certain design point 

60 1. Design point for each compressor train is automatically calculated from the input data such that the chart "just" 

61 cover the maximum rate/head in the input. 

62 2. Design point for each compressor stage is specified in the input. Useful to e.g. rerun the same compressor train 

63 for a different set of input data after first running with automatically calculated design points 

64 3. Compressor chart for each stage is specified in the input. Not yet implemented, will come later 

65 The compressor chart generator object, is an object which returns a compressor chart for each stage based on the 

66 compressor chart model chosen. 

67 

68 FluidStream: 

69 Model of the fluid. See FluidStream 

70 

71 Compressor inlet temperature: 

72 As a simplification, it is assumed that all compressor stages has the same inlet temperature (e.g. that there is an 

73 inter-stage cooling which always cool to the same temperature and that this is also equal at inlet of the first 

74 compressor). 

75 

76 This class is meant as a template class, and the sub types for each compressor chart model is a subclass of this 

77 - Use CompressorTrainSimplifiedChartsEstimated for a compressor chart model where design points are automatically 

78 calculated from input data and used to scale the generic unified chart 

79 - Use CompressorTrainSimplifiedChartsFromDesignPoints for a compressor chart model where design points per stage are 

80 given and the generic unified chart is scaled by these. 

81 """

364 ) -> NDArray[np.float64]: 

365 """For each stage 

366 Setup streams for inlet and outlet 

367 Calculate z and kappa for inlet and outlet 

368 Set z and kappa equal to average of in and out 

369 Calculate polytropic head for z and kappa 

370 Find maximum actual volume rate given head for each stage 

371 Calculate corresponding standard rates for each stage 

372 Maximum rate is minimum of the maximum standard rates for all stages. 

373 """ 

374 if self.stages is None: 

375 raise ValueError("Can't calculate max pressure when compressor stages are not defined.") 

376 

377 use_stage_for_maximum_rate_calculation = np.asarray([stage.compressor_chart for stage in self.stages]) 

378 if not use_stage_for_maximum_rate_calculation.any(): 

379 msg = "Calculating maximum rate is not possible for when all compressor charts are generic from data" 

380 logger.error(msg) 

381 return np.full_like(suction_pressures, fill_value=np.nan, dtype=float) 

382 

383 pressure_ratios_per_stage = self.calculate_pressure_ratios_per_stage( 

384 suction_pressure=suction_pressures, discharge_pressure=discharge_pressures 

385 ) 

386 inlet_pressure_all_stages = self._calulate_inlet_pressure_stages( 

387 inlet_pressure=suction_pressures, 

388 pressure_ratios_per_stage=pressure_ratios_per_stage, 

389 number_of_stages=len(self.stages), 

390 ) 

391 

392 # Take out data only for stages used for calculating maximum rate 

393 # Stages with compressor charts which are estimated from input data are not used, as these have per definition 

394 # infinite capacity as it can always be enlarged to account for the rate needed 

395 stages_to_use = [stage for stage in self.stages if stage.compressor_chart] 

396 inlet_pressure_stages_to_use = [ 

397 inlet_pressure 

398 for stage_number, inlet_pressure in enumerate(inlet_pressure_all_stages) 

399 if use_stage_for_maximum_rate_calculation[stage_number] 

400 ] 

401 inlet_temperatures_kelvin_to_use = [stage.inlet_temperature_kelvin for stage in stages_to_use] 

402 compressor_charts = [stage.compressor_chart for stage in stages_to_use] 

403 

404 stages_maximum_standard_rates = [ 

405 self.calculate_maximum_rate_given_outlet_pressure_all_calculation_points( 

406 inlet_pressure=inlet_pressure_stages_to_use[stage_number], 

407 compressor_chart=chart, 

408 pressure_ratio=pressure_ratios_per_stage, 

409 inlet_temperature_kelvin=np.full_like( 

410 inlet_pressure_stages_to_use[stage_number], inlet_temperatures_kelvin_to_use[stage_number]

460 inlet_stream=inlet_stream_calculation_point, 

461 molar_mass=fluid.molar_mass_kg_per_mol, 

462 compressor_chart=compressor_chart, 

463 initial_guess_head_at_maximum_actual_volume_rate=float( 

464 np.mean(compressor_chart.maximum_speed_curve.head_values) 

465 ), 

466 ) 

467 for inlet_pressure_calculation_point, pressure_ratio_calculation_point, inlet_temperature_kelvin_calculation_point, inlet_stream_calculation_point in zip( 

468 inlet_pressure, pressure_ratio, inlet_temperature_kelvin, inlet_streams 

469 ) 

470 ] 

471 inlet_densities_kg_per_m3 = [inlet_stream.density for inlet_stream in inlet_streams] 

472 maximum_mass_rates_kg_per_hour = [ 

473 actual_rate * density 

474 for actual_rate, density in zip(maximum_actual_rates_am3_per_hour, inlet_densities_kg_per_m3) 

475 ] 

476 maximum_standard_rates = fluid.mass_rate_to_standard_rate( 

477 mass_rate_kg_per_hour=np.array(maximum_mass_rates_kg_per_hour) 

478 ) 

479 

480 return maximum_standard_rates 

481 

482 @staticmethod 

483 def calculate_maximum_rate_given_outlet_pressure_single_calculation_point( 

484 inlet_pressure: float, 

485 pressure_ratio: float, 

486 inlet_temperature_kelvin: float, 

487 inlet_stream: FluidStream, 

488 molar_mass: float, 

489 compressor_chart: VariableSpeedCompressorChart, 

490 initial_guess_head_at_maximum_actual_volume_rate: float, 

491 ) -> float: 

492 outlet_pressure = inlet_pressure * pressure_ratio 

493 

494 if pressure_ratio < 1: 

495 error_message = "Outlet pressure ratio can not be less than 1" 

496 logger.error(error_message) 

497 raise IllegalStateException(error_message) 

498 

499 maximum_rate_function = compressor_chart.get_maximum_rate 

500 

501 polytropic_head = initial_guess_head_at_maximum_actual_volume_rate 

502 

503 z = inlet_stream.z 

504 kappa = inlet_stream.kappa 

505 

506 maximum_actual_volume_rate_previous = 1e-5 # Small but finite number to avoid division by zero. 

507 converged = False 

508 i = 0 

509 expected_diff = 1e-3 

510 max_iterations = 20 

511 while not converged and i < max_iterations: 

512 maximum_actual_volume_rate: float = float( 

513 maximum_rate_function( 

514 heads=polytropic_head, 

515 extrapolate_heads_below_minimum=False, 

516 ) 

517 ) 

518 

519 efficiency_array: NDArray[np.float64] = compressor_chart.efficiency_as_function_of_rate_and_head( 

520 rates=np.atleast_1d(maximum_actual_volume_rate), 

521 heads=np.atleast_1d(polytropic_head), 

522 ) 

523 polytropic_efficiency = efficiency_array[0] 

524 

525 polytropic_head = calculate_polytropic_head_campbell( 

526 polytropic_efficiency=polytropic_efficiency, 

527 kappa=kappa, 

528 z=z, 

529 molar_mass=molar_mass, 

530 pressure_ratios=pressure_ratio, 

531 temperatures_kelvin=inlet_temperature_kelvin, 

532 ) 

533 enthalpy_change_joule_per_kg = polytropic_head / polytropic_efficiency 

534 

535 outlet_stream = inlet_stream.set_new_pressure_and_enthalpy_change( 

536 new_pressure=outlet_pressure, enthalpy_change_joule_per_kg=enthalpy_change_joule_per_kg 

537 ) 

538 

539 # Set convergence criterion on actual volume rate 

540 diff = np.linalg.norm(maximum_actual_volume_rate - maximum_actual_volume_rate_previous) / np.linalg.norm( 

541 maximum_actual_volume_rate_previous 

542 ) 

543 converged = diff < expected_diff 

544 

545 # Update z and kappa estimates based on new outlet estimates 

546 z = (inlet_stream.z + outlet_stream.z) / 2.0 

547 kappa = (inlet_stream.kappa + outlet_stream.kappa) / 2.0 

548 maximum_actual_volume_rate_previous = maximum_actual_volume_rate 

549 

550 i += 1 

551 

552 if i == max_iterations: 

553 logger.error( 

554 "Simplified train: calculate_maximum_rate_given_outlet_pressure_single_calculation_point" 

555 f" did not converge after {max_iterations} iterations." 

556 f" inlet_pressure: {inlet_pressure}" 

557 f" pressure_ratio: {pressure_ratio}." 

558 f" inlet_temperature_kelvin: {inlet_temperature_kelvin}." 

559 f" molar_mass: {molar_mass}." 

560 f" Final difference between target and results was {diff}," 

561 f" while the convergence criterion is set to difference lower than {expected_diff}" 

562 f" NOTE! We will use the closest result we got for further calculations." 

563 " This should normally not happen. Please contact eCalc support." 

564 ) 

565 

566 return maximum_actual_volume_rate

569class CompressorTrainSimplifiedUnknownStages(CompressorTrainSimplified): 

570 """A simplified compressor train model where the number of compressors is not known before run-time 

571 Based on input data at run time evaluation, the number of stages is calculated based on maximum pressure ratio per 

572 stage. 

573 There is only one compressor chart given which is used for all stages. This chart may be a generic chart for which 

574 the design point is determined at run time given the input variables. 

575 

576 """ 

577 

578 def __init__( 

579 self, 

580 data_transfer_object: CompressorTrainSimplifiedWithUnknownStages, 

581 ): 

582 logger.debug("Creating CompressorTrainSimplifiedUnknownStages") 

583 super().__init__(data_transfer_object) 

584 self.data_transfer_object = data_transfer_object 

585 

586 def get_stages( 

587 self, 

588 rate: NDArray[np.float64], 

589 suction_pressure: NDArray[np.float64], 

590 discharge_pressure: NDArray[np.float64], 

591 ) -> list[CompressorTrainStage]: 

592 if len(rate) == 0: 

593 # Unable to figure out stages and pressure ratios if there are no rates as input 

594 return [] 

595 

596 pressure_ratios = discharge_pressure / suction_pressure 

597 maximum_pressure_ratio = max(pressure_ratios) 

598 number_of_compressors = self._calculate_number_of_compressors_needed( 

599 total_maximum_pressure_ratio=maximum_pressure_ratio, 

600 compressor_maximum_pressure_ratio=self.data_transfer_object.maximum_pressure_ratio_per_stage, 

601 ) 

602 

603 return [ 

604 map_compressor_train_stage_to_domain(self.data_transfer_object.stage) for _ in range(number_of_compressors) 

605 ]

607 @staticmethod 

608 def _calculate_number_of_compressors_needed( 

609 total_maximum_pressure_ratio: float, 

610 compressor_maximum_pressure_ratio: float, 

611 ) -> int: 

612 """Calculate min number of compressors given a maximum pressure ratio per compressor (default 3.5).""" 

613 x = math.log(total_maximum_pressure_ratio) / math.log(compressor_maximum_pressure_ratio) 

614 return math.ceil(x)

622 return np.full_like(suction_pressures, fill_value=np.nan, dtype=float)
