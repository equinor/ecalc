152    @staticmethod
153    def get_main_file(files: List[ValidEcalcFile]) -> ValidEcalcFile:
154        """Get the main yaml file. Detected by checking for a specific format. Only the main
155        yaml file can have !include and a certain set of sections.
156
157        * Only one main file is allowed
158
159        if none or more than one main file is detected an exception is raised
160        :param files:
161        :return:
162        """
163        main_files = [file for file in files if file.is_yaml(file.original_filename) and file.is_main_yaml_file()]
164        if len(main_files) > 1:
165            raise EcalcError(
166                "Bad Request",
167                f"Only one main file is supported, the following files were detected as main files: {', '.join([main_file.filename for main_file in main_files])}",
168                error_type=EcalcErrorType.CLIENT_ERROR,
169            )
170
171        if len(main_files) == 0:
172            raise EcalcError(
173                "Bad Request",
174                "No main files found. There must be one main file",
175                error_type=EcalcErrorType.CLIENT_ERROR,
176            )
177
178        return main_files[0]

180    @staticmethod
181    def validate_filetypes(files: List[FileWithName]) -> Tuple[List[ValidEcalcFile], List[InvalidEcalcFile]]:
182        """Given a list of files (e.g uploaded by a user), given a name, do an initial attempt to
183        filter out bad files for further processing of good files only.
184
185        The current validation implementation is very naĩve, by only checking file extension.
186
187        :param files:
188        :return:
189        """
190        valid_files: List[ValidEcalcFile] = []
191        invalid_files: List[InvalidEcalcFile] = []
192        for file in files:
193            if EcalcFiles.allowed_file(file.filename):
194                if EcalcFile.is_zip(Path(file.filename)):
195                    if len(files) > 1:
196                        raise EcalcError(
197                            title="Invalid file combination",
198                            message="A zip file cannot be combined with other file types. Please provide a zip file alone.",
199                        )
200
201                    valid_files, invalid_files = unpack_zip(file.file)
202                elif EcalcFile.is_csv(Path(file.filename)) or EcalcFile.is_yaml(Path(file.filename)):
203                    valid_files.append(
204                        ValidEcalcFile(
205                            original_filename=Path(file.filename),
206                            filename=file.filename,
207                            file=file.file,
208                            file_type=EcalcFile.get_type(filename=Path(file.filename)),
209                        )
210                    )
211                else:
212                    invalid_files.append(
213                        InvalidEcalcFile(
214                            original_filename=Path(file.filename),
215                            filename=file.filename,
216                            error="Invalid File Extension",
217                        )
218                    )
219            else:
220                invalid_files.append(
221                    InvalidEcalcFile(
222                        original_filename=Path(file.filename),
223                        filename=file.filename,
224                        error="Invalid File Extension",
225                    )
226                )
227
228        return valid_files, invalid_files

231def find_longest_common_path(file_path_1: Path, file_path_2: Path) -> str:
232    """Given 2 paths, find the longest common path, part by part that the 2 paths share; ie
233    until which position in the file hierarchy do they diverge?
234
235    :param file_path_1:
236    :param file_path_2:
237    :return:
238    """
239    common_path = ""
240    for path_1, path_2 in zip(file_path_1.parts, file_path_2.parts):
241        if path_1 == path_2:
242            common_path += path_1 + "/"
243        else:
244            break
245
246    return common_path

258def make_relative_path(linked_file: str, main_file: str) -> str:
259    """Given a main file and a file to be linked to from that file, only
260    include the parts of the path of both files that _differ_.
261
262    The parts that is different in main_file, will be replaced with "../",
263
264    :param linked_file:
265    :param main_file:
266    :return:
267    """
268    main_file_path = Path(main_file)
269    for part in main_file_path.parts[:-1]:
270        if part != "" and part != "/":
271            linked_file = "../" + linked_file
272
273    return linked_file

288def rename_duplicates(valid_files: List[ValidEcalcFile], duplicates: List[str]) -> Dict[Path, str]:
289    """Rename duplicate files. All with same name will be renamed. Those that are not duplicates,
290    will also be returned, with the original filename in the mapping.
291
292    Only resource files (csv) needs to be renamed, because they are flattened out and will then exist
293    at the same level.
294
295    :param valid_files:
296    :param duplicates:
297    :return:
298    """
299    renamed_files: Dict[Path, str] = {}
300    for file in valid_files:
301        if file.filename in duplicates:
302            renamed_filename = str(file.original_filename).replace("/", "_")
303            if len(renamed_filename) > 100:
304                renamed_filename = renamed_filename[-94:] + str(uuid.uuid4().hex.upper()[0:6])
305            renamed_files[file.original_filename] = renamed_filename
306        else:
307            renamed_files[file.original_filename] = file.filename
308
309    return renamed_files

312def make_relative_paths(files: List[ValidEcalcFile], main_yaml: ValidEcalcFile) -> Dict[Path, str]:
313    """For all files in the list, generate the relative paths for all files, relative
314    to the provided main file. All files provided, must have the original_filename set, which must
315    be the relative path to the root of the model; e.g. the (zip) archive path, or
316    if only one level, the filename itself.
317
318    :param files:
319    :param main_yaml:
320    :return:
321    """
322    relative_paths: Dict[Path, str] = {}
323    for file in files:
324        if file == main_yaml:
325            # since all files are relative to this given file, this must be the filename itself, only
326            relative_paths[main_yaml.original_filename] = main_yaml.filename
327        else:
328            common_path = find_longest_common_path(Path(file.original_filename), Path(main_yaml.original_filename))
329            stripped_file = strip_common_path(common_path, str(file.original_filename))
330            stripped_main = strip_common_path(common_path, str(main_yaml.original_filename))
331            relative_path = make_relative_path(stripped_file, stripped_main)
332
333            relative_paths[file.original_filename] = relative_path
334
335    return relative_paths

338def unpack_zip(file: IO) -> Tuple[List[ValidEcalcFile], List[InvalidEcalcFile]]:
339    """Unpack the zip similarility to how single files are handled, by returning a tuple
340    of valid and invalid files.
341
342    :param file:
343    :return:
344    """
345    valid_files: List[ValidEcalcFile] = []
346    invalid_files: List[InvalidEcalcFile] = []
347    try:
348        with zipfile.ZipFile(BytesIO(file.read())) as archive:
349            for zip_info in archive.infolist():
350                try:
351                    file_path = Path(zip_info.filename)
352
353                    if is_ignored_zip_content(file_path) or zip_info.is_dir():
354                        # Just ignore, dont event mention it. Will just confuse users (because those files are normally hidden)
355                        continue
356                    elif EcalcFile.is_csv(file_path) or EcalcFile.is_yaml(file_path):
357                        with archive.open(zip_info) as file:
358                            file_like = TemporaryFile()
359                            file_like.write(file.read())
360                            file_like.seek(0)
361                            valid_files.append(
362                                ValidEcalcFile(
363                                    original_filename=file_path,
364                                    filename=file_path.name,
365                                    file=file_like,
366                                    file_type=EcalcFile.get_type(filename=file_path),
367                                )
368                            )
369                    else:
370                        invalid_files.append(
371                            InvalidEcalcFile(
372                                original_filename=file_path, filename=file_path.name, error="Invalid file extension"
373                            )
374                        )
375                        continue
376
377                except EcalcError as ee:
378                    logger.warning(f"An error occurred while reading file({file_path}) in zip archive")
379                    invalid_files.append(
380                        InvalidEcalcFile(original_filename=file_path, filename=file_path.name, error=ee.message)
381                    )
382
383        valid_file_paths = [valid_file.original_filename for valid_file in valid_files]
384        if len(valid_file_paths) != len(set(valid_file_paths)):
385            raise EcalcError(title="Bad zip file", message="Duplicated filepaths in zip archive detected. Please fix.")
386
387        return valid_files, invalid_files
388
389    except zipfile.BadZipFile as e:
390        raise EcalcError(title="Bad zip file", message="An error occurred while unpacking the zip file") from e

393def _validate_headers(headers: List[str]):
394    for header in headers:
395        if not re.match(r"^[A-Za-z][A-Za-z0-9_.,\-\s#+:\/]*$", header):
396            raise ValueError(
397                "Each header value must start with a letter in the english "
398                "alphabet (a-zA-Z). And may only contain letters, spaces, numbers or any of the following characters "
399                "[ _ - # + : . , /] "
400            )
401        elif re.match(r"^Unnamed: \d+$", header):
402            raise InvalidResourceHeaderException(
403                "One or more headers are missing in time series or facilities resource"
404            )

407def _validate_not_nan(columns: List[List]):
408    for column in columns:

411                raise ValueError(
412                    f"csv file contains invalid data at row {index + 1}, "
413                    f"all headers must be associated with a valid column value"

417def _dataframe_to_resource(df: pd.DataFrame, validate_headers: bool = True) -> Resource:

470def read_timeseries_resource(
471    resource_input: Union[Path, BytesIO, str],
472    timeseries_type: YamlTimeseriesType,
473    validate_headers: bool = True,
474) -> Resource:
475    """Read timeseries resource from filepath with timeseries specific manipulation/validation.
476
477    - Timeseries is allowed to have nans
478    """
479    if not isinstance(resource_input, (BytesIO, str, Path)):
480        raise ValueError(f"Invalid resource_input type '{type(resource_input)}'")
481
482    if timeseries_type in (YamlTimeseriesType.DEFAULT, YamlTimeseriesType.MISCELLANEOUS):
483        if isinstance(resource_input, Path):
484            with open(resource_input) as resource_file:
485                resource_df = read_csv(resource_file)
486        else:
487            resource_df = read_csv(resource_input)
488    else:
489        raise ValueError(f"Invalid timeseries type '{timeseries_type}' for resource '{resource_input}'")
490
491    headers = resource_df.columns.tolist()
492    headers = [header.strip() for header in headers]
493    if validate_headers:
494        _validate_headers(headers)
495    return convert_dataframe_to_timeseries_resource(resource_df=resource_df)

500
501    - Facility files are not allowed to have nans

Such classes are primarily used with static type checkers that recognize structural subtyping (static duck-typing), for example::

339def unpack_zip(file: IO) -> Tuple[List[ValidEcalcFile], List[InvalidEcalcFile]]:
340    """Unpack the zip similarility to how single files are handled, by returning a tuple
341    of valid and invalid files.
342
343    :param file:
344    :return:
345    """
346    valid_files: List[ValidEcalcFile] = []
347    invalid_files: List[InvalidEcalcFile] = []
348    try:
349        with zipfile.ZipFile(BytesIO(file.read())) as archive:
350            for zip_info in archive.infolist():
351                try:
352                    file_path = Path(zip_info.filename)
353
354                    if is_ignored_zip_content(file_path) or zip_info.is_dir():
355                        # Just ignore, dont event mention it. Will just confuse users (because those files are normally hidden)
356                        continue
357                    elif EcalcFile.is_csv(file_path) or EcalcFile.is_yaml(file_path):
358                        with archive.open(zip_info) as file:
359                            file_like = TemporaryFile()
360                            file_like.write(file.read())
361                            file_like.seek(0)

365                                    filename=file_path.name,
366                                    file=file_like,

368                                )
369                            )
370                    else:
371                        invalid_files.append(
372                            InvalidEcalcFile(

374                            )
375                        )
376                        continue
377

379                    logger.warning(f"An error occurred while reading file({file_path}) in zip archive")

381                        InvalidEcalcFile(original_filename=file_path, filename=file_path.name, error=ee.message)
382                    )

386            raise EcalcError(title="Bad zip file", message="Duplicated filepaths in zip archive detected. Please fix.")
