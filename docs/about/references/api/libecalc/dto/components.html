450    def validate_unique_names(self):
451        """Ensure unique component names within installation."""
452        names = [self.name]
453        fuel_types = [dto.FuelType]
454        fuel_names = [str]
455        for installation in self.installations:

460            names.extend([venting_emitter.name for venting_emitter in venting_emitters])
