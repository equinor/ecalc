 62def _subtract_list(first: List[Optional[float]], second: List[Optional[float]]):
 63    subtracted = []
 64    for f, s in zip(first, second):
 65        if f is None:
 66            f = 0
 67        if s is None:
 68            s = 0
 69        subtracted.append(f - s)
 70
 71    return subtracted

 74class InterpolationMethod(str, enum.Enum):
 75    SKIP = "SKIP"
 76    LINEAR = "LINEAR"

 81    InterpolationMethod.LINEAR: "linear",
 82}

 85def _interpolate_list(
 86    x: List[datetime],
 87    y: List[float],
 88    target_x: List[datetime],
 89    method: InterpolationMethod,
 90) -> List[float]:
 91    if len(y) != len(x):
 92        raise ValueError("x and y should have equal length.")
 93
 94    if len(y) == 0:
 95        return []
 96
 97    if len(y) == 1:
 98        return [y[0]] * len(target_x)
 99
100    if x == target_x:
101        return y
102
103    if len(target_x) == 0:
104        return []
105
106    kind = interpolationKindMap[method]
107    start = min(x[0], target_x[0])
108    x_in_seconds = [(time_step - start).total_seconds() for time_step in x]
109    target_x_in_seconds = [(time_step - start).total_seconds() for time_step in target_x]
110
111    return list(
112        interp1d(
113            x=x_in_seconds,
114            y=y,
115            fill_value=(y[0], y[-1]),
116            assume_sorted=True,
117            kind=kind,
118        )(target_x_in_seconds)
119    )

162    @classmethod
163    def fit_to_timesteps(
164        cls,
165        component: "SimpleComponentResult",
166        timesteps: List[datetime],
167        method: InterpolationMethod = InterpolationMethod.SKIP,
168    ) -> "SimpleComponentResult":
169        """Fit the component to the provided timesteps based on specified method.
170
171        :param component: The component that should be normalized
172        :param timesteps: the target timesteps. These should already be correct based on the InterpolationMethod
173            specified, i.e. if InterpolationMethod is SKIP, the provided timesteps should all exist in the component.
174        :param method: The interpolation method that should be used to fill values
175        :return: the normalized component
176        """
177        if method == InterpolationMethod.SKIP:
178            power = []
179            energy_usage = []
180            is_valid = []
181            emissions: Dict[str, SimpleEmissionResult] = {
182                emission.name: SimpleEmissionResult(name=emission.name, rate=[])
183                for emission in component.emissions.values()
184            }
185            for timestep in timesteps:
186                if timestep in component.timesteps:
187                    timestep_index = component.timesteps.index(timestep)
188
189                    if component.power is not None:
190                        power.append(component.power[timestep_index])
191
192                    energy_usage.append(component.energy_usage[timestep_index])
193
194                    is_valid.append(component.is_valid[timestep_index])
195                    for emission in emissions.values():
196                        # Assume index exist if emission exist
197                        emission.rate.append(component.emissions[emission.name].rate[timestep_index])
198                else:
199                    # This is a developer error, we should provide the correct timesteps.
200                    raise ValueError(
201                        f"Provided timesteps includes timestep not found in component {component.id}. "
202                        f"Extraneous timestep: {timestep}. This should not happen, contact support."
203                    )
204        elif method == InterpolationMethod.LINEAR:
205            emissions = {
206                emission.name: SimpleEmissionResult(
207                    name=emission.name,
208                    rate=_interpolate_list(
209                        x=component.timesteps,
210                        y=emission.rate,
211                        target_x=timesteps,
212                        method=method,
213                    ),
214                )
215                for emission in component.emissions.values()
216            }
217
218            is_valid = []
219            for timestep in timesteps:
220                if timestep in component.timesteps:
221                    timestep_index = component.timesteps.index(timestep)
222                    is_valid.append(component.is_valid[timestep_index])
223                else:
224                    is_valid.append(True)
225
226            power = None
227            if component.power is not None:
228                power = _interpolate_list(
229                    x=component.timesteps,
230                    y=component.power,
231                    target_x=timesteps,
232                    method=method,
233                )
234
235            energy_usage = _interpolate_list(
236                x=component.timesteps,
237                y=component.energy_usage,
238                target_x=timesteps,
239                method=method,
240            )
241        else:
242            raise NotImplementedError()
243
244        return cls(
245            componentType=component.componentType,
246            component_level=component.component_level,
247            parent=component.parent,
248            name=component.name,
249            power=power,
250            energy_usage=energy_usage,
251            energy_usage_unit=component.energy_usage_unit,
252            emissions=emissions,
253            is_valid=is_valid,
254            timesteps=timesteps,
255        )

257    def __sub__(self, reference_component) -> "SimpleComponentResult":
258        """Calculate difference between two component results."""
259        if not isinstance(reference_component, SimpleComponentResult):
260            raise TypeError(
261                f"unsupported operand type(s) for -: '{type(self).__name__}' and '{type(reference_component).__name__}'"
262            )
263        if self.id != reference_component.id:
264            raise ValueError(
265                f"Can not compare different components, id does not match: {self.id} and {reference_component.id}"
266            )
267        if self.timesteps != reference_component.timesteps:
268            raise ValueError(
269                f"Can not compare components with differing timesteps: {self.id} and {reference_component.id}"
270            )
271        if self.energy_usage_unit != reference_component.energy_usage_unit:
272            raise ValueError(
273                f"Can not compare components with differing energy usage units: "
274                f"{self.id} with unit '{self.energy_usage_unit}' and "
275                f"{reference_component.id} with unit '{reference_component.energy_usage_unit}'."
276            )
277
278        return SimpleComponentResult(
279            name=self.name,
280            parent=self.parent if self.parent == reference_component.parent else None,
281            componentType=self.componentType,
282            component_level=self.component_level,
283            timesteps=self.timesteps,
284            is_valid=[changed and reference for changed, reference in zip(self.is_valid, reference_component.is_valid)],
285            energy_usage=_subtract_list(self.energy_usage, reference_component.energy_usage),
286            energy_usage_unit=self.energy_usage_unit,
287            power=_subtract_list(self.power, reference_component.power),  # type: ignore[arg-type]
288            emissions={
289                self_e.name: self_e - reference_e
290                for self_e, reference_e in zip(
291                    sorted(self.emissions.values(), key=lambda emission: emission.name),
292                    sorted(reference_component.emissions.values(), key=lambda emission: emission.name),
293                )
294            },
295        )

338    @classmethod
339    def subtract(
340        cls,
341        other_model: "SimpleResultData",
342        reference_model: "SimpleResultData",
343    ) -> Tuple["SimpleResultData", List[Exception]]:
344        """Subtract reference model from other model.
345
346        Timesteps and components should be equal between the models.
347        """
348        if other_model.timesteps != reference_model.timesteps:
349            raise ValueError(
350                "Timesteps should be equal between models when calculating delta profile. "
351                "Use separate methods to normalize."
352            )
353        timesteps = reference_model.timesteps
354
355        errors = []
356        components = []
357        for other_component, reference_component in zip(other_model.components, reference_model.components):
358            try:
359                components.append(other_component - reference_component)
360            except ValueError as e:
361                logger.error(e)
362                errors.append(e)
363
364        return SimpleResultData(timesteps=timesteps, components=components), errors

366    @classmethod
367    def _normalize_emissions(cls, other_component, reference_component):
368        emission_names = set(other_component.emissions.keys()).union(x for x in reference_component.emissions)
369
370        for emission_name in emission_names:
371            for component in [other_component, reference_component]:
372                if emission_name not in list(component.emissions):
373                    vector_length = len(component.timesteps)
374                    component.emissions[emission_name] = SimpleEmissionResult(
375                        name=emission_name,
376                        rate=[0] * vector_length,
377                    )
378
379        return other_component, reference_component

381    @classmethod
382    def normalize_components(
383        cls,
384        other_model: "SimpleResultData",
385        reference_model: "SimpleResultData",
386        exclude: Optional[List[ComponentType]] = None,
387    ):
388        if exclude is None:
389            exclude = []
390
391        other_components = {component.id: component for component in other_model.components}
392        reference_components = {component.id: component for component in reference_model.components}
393
394        component_ids = sorted(set(other_components.keys()).union(set(reference_components.keys())))
395
396        filtered_component_ids = [
397            component_id for component_id in component_ids if component_id.componentType not in exclude
398        ]
399
400        normalized_reference_components = []
401        normalized_other_components = []
402
403        for component_id in filtered_component_ids:
404            other_component = other_components.get(component_id)
405            reference_component = reference_components.get(component_id)
406
407            if other_component is None and reference_component is not None:
408                other_component = _create_empty_component(reference_component)
409            elif reference_component is None and other_component is not None:
410                reference_component = _create_empty_component(other_component)
411
412            other_component, reference_component = cls._normalize_emissions(other_component, reference_component)
413
414            normalized_reference_components.append(reference_component)
415            normalized_other_components.append(other_component)
416
417        return (
418            cls(timesteps=other_model.timesteps, components=normalized_other_components),
419            cls(timesteps=reference_model.timesteps, components=normalized_reference_components),
420        )

422    @classmethod
423    def delta_profile(
424        cls,
425        other_model: "SimpleResultData",
426        reference_model: "SimpleResultData",
427        interpolation_method: InterpolationMethod = InterpolationMethod.SKIP,
428    ) -> Tuple["SimpleResultData", "SimpleResultData", "SimpleResultData", List[str]]:
429        if interpolation_method == InterpolationMethod.SKIP:
430            timesteps = sorted(set(other_model.timesteps).intersection(reference_model.timesteps))
431        elif interpolation_method == InterpolationMethod.LINEAR:
432            timesteps = sorted(set(other_model.timesteps).union(reference_model.timesteps))
433        else:
434            raise NotImplementedError
435
436        # Normalize components first as we need to filter out CONSUMER_MODELs for normalize_timesteps to work.
437        other_model, reference_model = cls.normalize_components(other_model, reference_model)
438
439        other_model = cls.fit_to_timesteps(other_model, timesteps, method=interpolation_method)
440        reference_model = cls.fit_to_timesteps(reference_model, timesteps, method=interpolation_method)
441
442        delta_profile, errors = cls.subtract(other_model, reference_model)
443
444        return other_model, reference_model, delta_profile, [str(error) for error in errors]

171

173        :param timesteps: the target timesteps. These should already be correct based on the InterpolationMethod

175        :param method: The interpolation method that should be used to fill values
176        :return: the normalized component

178        if method == InterpolationMethod.SKIP:
179            power = []
180            energy_usage = []
181            is_valid = []
182            emissions: Dict[str, SimpleEmissionResult] = {
183                emission.name: SimpleEmissionResult(name=emission.name, rate=[])
184                for emission in component.emissions.values()
185            }
186            for timestep in timesteps:
187                if timestep in component.timesteps:
188                    timestep_index = component.timesteps.index(timestep)
189
190                    if component.power is not None:

192
193                    energy_usage.append(component.energy_usage[timestep_index])
194
195                    is_valid.append(component.is_valid[timestep_index])
196                    for emission in emissions.values():
197                        # Assume index exist if emission exist
198                        emission.rate.append(component.emissions[emission.name].rate[timestep_index])
199                else:
200                    # This is a developer error, we should provide the correct timesteps.
201                    raise ValueError(
202                        f"Provided timesteps includes timestep not found in component {component.id}. "
203                        f"Extraneous timestep: {timestep}. This should not happen, contact support."
204                    )
205        elif method == InterpolationMethod.LINEAR:
206            emissions = {
207                emission.name: SimpleEmissionResult(
208                    name=emission.name,
209                    rate=_interpolate_list(

212                        target_x=timesteps,
213                        method=method,
214                    ),
215                )
216                for emission in component.emissions.values()
217            }
218
219            is_valid = []
220            for timestep in timesteps:
221                if timestep in component.timesteps:

223                    is_valid.append(component.is_valid[timestep_index])
224                else:

230                    x=component.timesteps,

237                x=component.timesteps,

239                target_x=timesteps,
240                method=method,
241            )
242        else:
243            raise NotImplementedError()
244
245        return cls(

247            component_level=component.component_level,
