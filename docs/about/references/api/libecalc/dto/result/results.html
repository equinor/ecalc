163class EcalcModelResult(EcalcResultBaseModel):
164    """Result object holding one component for each part of the eCalc model run:
165
166    ModelResult, InstallationResult, GeneratorSetResult, ConsumerSystemResult, ConsumerGroupResult and ConsumerResult
167    """
168
169    component_result: ComponentResult
170    # Setting min and max items to be able to generate OpenAPI:
171    # Ref. https://github.com/developmentseed/geojson-pydantic/issues/42
172    sub_components: Annotated[List[ComponentResult], Field(min_items=0, max_items=10000)]
173    models: Annotated[List[ConsumerModelResult], Field(min_items=0, max_items=10000)]
174
175    @validator("sub_components")
176    def sort_sub_components(cls, sub_components):
177        return sorted(sub_components, key=attrgetter("componentType", "name"))
178
179    @validator("models")
180    def sort_models(cls, models):
181        return sorted(models, key=attrgetter("componentType", "name"))
182
183    @property
184    def timesteps(self):
185        return self.component_result.timesteps
186
187    @property
188    def components(self):
189        return [self.component_result, *self.sub_components]
190
191    def get_components(self, component_ids: List[str]) -> List[ComponentResult]:
192        return [component for component in self.components if component.id in component_ids]
193
194    def get_component_by_name(self, component_name: str) -> Optional[ComponentResult]:
195        components = [component for component in self.components if component.name == component_name]
196        if not components:
197            return None
198
199        if len(components) > 1:
200            logger.warning(f"Querying duplicate component {component_name}. Returning first match")
201
202        return components[0]
203
204    def simple_result(self):
205        return SimpleResultData(
206            timesteps=self.timesteps, components=[component.simple_result() for component in self.components]
207        )
208
209    def resample(self, freq: Frequency) -> EcalcModelResult:
210        return self.__class__(
211            component_result=self.component_result.resample(freq),
212            sub_components=[sub_component.resample(freq) for sub_component in self.sub_components],
213            models=[model.resample(freq) for model in self.models],
214        )
