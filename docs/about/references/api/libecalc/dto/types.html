117class FuelType(EcalcBaseModel):
118    """An installation/model/component may change fuel over time, due to setup changes,
119    production, and the cost may also change on different predictive models.
120    """
121
122    name: str
123    user_defined_category: Optional[FuelTypeUserDefinedCategoryType] = None
124    price: Optional[Expression]
125    emissions: List[Emission] = Field(default_factory=list)
126
127    _convert_expression = validator("price", allow_reuse=True, pre=True)(convert_expression)
128
129    @validator("price", pre=True)
130    def convert_price(cls, price):
131        # NOTE: This is called after validator/converter above, hence wraps value in an Expression
132        # This is needed when price is explicitly set to None, e.g. when parsed in YAML
133        return price if price is not None else Expression.setup_from_expression(value=0.0)
134
135    @validator("user_defined_category", pre=True, always=True)
136    def check_user_defined_category(cls, user_defined_category, values):
137        """Provide which value and context to make it easier for user to correct wrt mandatory changes."""
138        if user_defined_category is not None:
139            if user_defined_category not in list(FuelTypeUserDefinedCategoryType):
140                name = ""
141                if values.get("name") is not None:
142                    name = f"with the name {values.get('name')}"
143
144                raise ValueError(
145                    f"CATEGORY: {user_defined_category} is not allowed for {cls.__name__} {name}. Valid categories are: {[str(fuel_type_user_defined_category.value) for fuel_type_user_defined_category in FuelTypeUserDefinedCategoryType]}"
146                )
147
148        return user_defined_category
