489    @model_validator(mode="after")
490    def validate_unique_names(self):
491        """Ensure unique component names within installation."""
492        names = [self.name]
493        fuel_types = [FuelType]
494        fuel_names = [str]
495        for installation in self.installations:
496            names.append(installation.name)
497            fuel_consumers = installation.fuel_consumers
498            venting_emitters = installation.venting_emitters
499
500            names.extend([venting_emitter.name for venting_emitter in venting_emitters])
501            for fuel_consumer in fuel_consumers:
502                names.append(fuel_consumer.name)
503                if isinstance(fuel_consumer, GeneratorSet):
504                    for electricity_consumer in fuel_consumer.consumers:
505                        if isinstance(electricity_consumer, ConsumerSystem):
506                            for consumer in electricity_consumer.consumers:
507                                names.append(consumer.name)
508                elif isinstance(fuel_consumer, ConsumerSystem):
509                    for consumer in fuel_consumer.consumers:
510                        names.append(consumer.name)
511                if fuel_consumer.fuel is not None:
512                    for fuel_type in fuel_consumer.fuel.values():
513                        # Need to verify that it is a different fuel
514                        if fuel_type is not None and fuel_type not in fuel_types:
515                            fuel_types.append(fuel_type)
516                            fuel_names.append(fuel_type.name)
517
518        duplicated_names = get_duplicates(names)
519        duplicated_fuel_names = get_duplicates(fuel_names)
520
521        if len(duplicated_names) > 0:
522            raise ValueError(
523                "Component names must be unique. Components include the main model, installations,"
524                " generator sets, electricity consumers, fuel consumers, systems and its consumers and direct emitters."
525                f" Duplicated names are: {', '.join(duplicated_names)}"
526            )
527
528        if len(duplicated_fuel_names) > 0:
529            raise ValueError(
530                "Fuel type names must be unique across installations."
531                f" Duplicated names are: {', '.join(duplicated_fuel_names)}"
532            )
533        return self
