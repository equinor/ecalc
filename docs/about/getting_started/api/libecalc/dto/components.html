774class FuelModel:
775    """A function to evaluate fuel related attributes for different time period
776    For each period, there is a data object with expressions for fuel related
777    attributes which may be evaluated for some variables and a fuel_rate.
778    """
779
780    def __init__(self, fuel_time_function_dict: dict[Period, FuelType]):
781        logger.debug("Creating fuel model")
782        self.temporal_fuel_model = fuel_time_function_dict
783
784    def evaluate_emissions(
785        self, expression_evaluator: ExpressionEvaluator, fuel_rate: list[float]
786    ) -> dict[str, EmissionResult]:
787        """Evaluate fuel related expressions and results for a TimeSeriesCollection and a
788        fuel_rate array.
789
790        First the fuel parameters are calculated by evaluating the fuel expressions and
791        the time_series object.
792
793        Then the resulting emission volume is calculated based on the fuel rate:
794        - emission_rate = emission_factor * fuel_rate
795
796        This is done per time period and all fuel related results both in terms of
797        fuel types and time periods, are merged into one common fuel collection results object.
798
799        The length of the fuel_rate array must equal the length of the global list of periods.
800        It is assumed that the fuel_rate array origins from calculations based on the same time_series
801        object and thus will have the same length when used in this method.
802        """
803        logger.debug("Evaluating fuel usage and emissions")
804
805        fuel_rate = np.asarray(fuel_rate)
806
807        # Creating a pseudo-default dict with all the emitters as keys. This is to handle changes in a temporal model.
808        emissions = {
809            emission_name: EmissionResult.create_empty(name=emission_name, periods=Periods([]))
810            for emission_name in {
811                emission.name for _, model in self.temporal_fuel_model.items() for emission in model.emissions
812            }
813        }
814
815        for temporal_period, model in self.temporal_fuel_model.items():
816            if Period.intersects(temporal_period, expression_evaluator.get_period()):
817                start_index, end_index = temporal_period.get_period_indices(expression_evaluator.get_periods())
818                variables_map_this_period = expression_evaluator.get_subset(
819                    start_index=start_index,
820                    end_index=end_index,
821                )
822                fuel_rate_this_period = fuel_rate[start_index:end_index]
823                for emission in model.emissions:
824                    factor = variables_map_this_period.evaluate(expression=emission.factor)
825
826                    emission_rate_kg_per_day = fuel_rate_this_period * factor
827                    emission_rate_tons_per_day = Unit.KILO_PER_DAY.to(Unit.TONS_PER_DAY)(emission_rate_kg_per_day)
828
829                    result = EmissionResult(
830                        name=emission.name,
831                        periods=variables_map_this_period.get_periods(),
832                        rate=TimeSeriesStreamDayRate(
833                            periods=variables_map_this_period.get_periods(),
834                            values=emission_rate_tons_per_day.tolist(),
835                            unit=Unit.TONS_PER_DAY,
836                        ),
837                    )
838
839                    emissions[emission.name].extend(result)
840
841                for name in emissions:
842                    if name not in [emission.name for emission in model.emissions]:
843                        emissions[name].extend(
844                            EmissionResult.create_empty(name=name, periods=variables_map_this_period.get_periods())
845                        )
846
847        return dict(sorted(emissions.items()))
