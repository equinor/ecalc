 34    def __contains__(self, date_or_period: Union[datetime, Period]) -> bool:
 35        """
 36        A period of time is defined as [start, end>, ie inclusive start and exclusive end.
 37
 38        Check if a date or another period is contained within this period.

 41            date_or_period: A date or period of interest
 42
 43        Returns:
 44            Whether the given date or period is contained within this period

136        """
137        Create periods from the provided datetimes
138        :param times: the sorted times to create periods from

216def define_time_model_for_period(time_model_data: Optional[Any], target_period: Period) -> Optional[Dict[Period, Any]]:
217    """Process time model based on the target period.
218
219    Steps:
220        - Add a default start date if the model is not already a time model
221        - Filter definitions outside given time period
222        - Adjust start date of the first model to the period
223
224    Args:
225        time_model_data: A model that can vary based on time, e.g., {1900-01-01: some model, 1950-01-01: some other model}
226        target_period: Period for which a model is defined (START, END specified by user or default to everything)
227
228    Returns:
229        The time model for the target period
230    """
231    if time_model_data is None:
232        return None
233
234    # Make sure the model is a time model
235    time_model_data = default_temporal_model(time_model_data, default_period=target_period)
236
237    dict_to_return = {}
238    for model_period, model in time_model_data.items():
239        if Period.intersects(model_period, target_period):
240            dict_to_return[Period.intersection(model_period, target_period)] = model
241
242    return dict_to_return

302def create_start_and_end_dates_for_periods(
303    frequency: Frequency, start: datetime, end: datetime, include_start_date: bool, include_end_date: bool
304) -> List[datetime]:
305    """
306
307    Args:
308        frequency: The requested frequency
309        start: The start date
310        end: The end date
311        include_start_date: Whether to include the start date if it is not part of the requested frequency
312        include_end_date:  Whether to include the end date if it is not part of the requested frequency
313
314    Returns:
315        A list of dates (and possibly including the start/end dates) between the given start and end dates following
316        the requested frequency
317
318    """
319    # If the start date or end date is part of the date_range made by the frequency, the returned date range will
320    # always include the start and end date (no matter what the include_start_date and include_end_date booleans are).
321    # To avoid this add one day to start and subtract one day from end.
322    date_range = pd.date_range(start=start + timedelta(days=1), end=end - timedelta(days=1), freq=frequency.value)
323
324    time_steps = [clear_time(time_step) for time_step in date_range]
325    if include_start_date:
326        time_steps = [clear_time(start)] + time_steps
327    if include_end_date:
328        time_steps = [clear_time(end)] + time_steps
329
330    return sorted(set(time_steps))

337def is_temporal_model(data: Dict) -> bool:
338    """Check if the data is a time dependent dict. A time dependent dict is a dict where the keys are dates or periods."""
339    if isinstance(data, dict):
340        is_period = []
341        is_date = []
342        is_not_period_keys = []
343        for key in data:
344            if isinstance(key, date):
345                is_date.append(True)
346                is_period.append(False)
347            elif isinstance(key, Period):
348                is_period.append(True)
349                is_date.append(False)
350            else:
351                try:
352                    split_key = key.split(";")
353                    if len(split_key) == 1:
354                        datetime.strptime(split_key[0], "%Y-%m-%dT%H:%M:%S")
355                        is_date.append(True)
356                        is_period.append(False)
357                    else:
358                        datetime.strptime(split_key[0], "%Y-%m-%d %H:%M:%S")
359                        datetime.strptime(split_key[1], "%Y-%m-%d %H:%M:%S")
360                        is_period.append(True)
361                        is_date.append(False)
362                except (TypeError, ValueError):
363                    is_not_period_keys.append(str(key))
364                    is_period.append(False)
365                    is_date.append(False)
366        if any(is_date):
367            if not all(is_date):
368                raise InvalidDateException(
369                    title="Invalid model",
370                    message="Temporal models should only contain date keys. "
371                    f"Invalid date(s): {','.join(is_not_period_keys)}",
372                )
373            return True
374        if any(is_period):
375            if not all(is_period):
376                raise InvalidDateException(
377                    title="Invalid model",
378                    message="Temporal models should only contain date keys. "
379                    f"Invalid date(s): {','.join(is_not_period_keys)}",
380                )
381            return True
382    return False

391def default_temporal_model(data: Any, default_period: Period) -> Optional[Dict[Period, Any]]:
392    """Ensure the data is a time dependent dict. Also convert all dates to datetime with default time 00:00:00
393    :param default_period: the period to use as default
394    :param data:
395    :return:
396    """
397    if data is None:
398        return None
399    elif is_temporal_model(data):
400        # Already a temporal model dictionary. Convert all keys to periods if they are dates
401        if isinstance(next(iter(data)), date):
402            _dates = [convert_date_to_datetime(key) for key in data.keys()] + [datetime.max.replace(microsecond=0)]
403            return {
404                Period(start=start_date, end=end_date): value
405                for start_date, end_date, value in zip(_dates[:-1], _dates[1:], data.values())
406            }
407        return {
408            Period(start=convert_date_to_datetime(key.start), end=convert_date_to_datetime(key.end)): value
409            for key, value in data.items()
410        }
411    else:
412        # Set default start
413        return {
414            default_period: data,
415        }

338def is_temporal_model(data: Dict) -> bool:
339    """Check if the data is a time dependent dict. A time dependent dict is a dict where the keys are dates or periods."""
340    if isinstance(data, dict):
341        is_period = []
342        is_date = []
343        is_not_period_keys = []
344        for key in data:
345            if isinstance(key, date):
346                is_date.append(True)
347                is_period.append(False)
348            elif isinstance(key, Period):
349                is_period.append(True)
350                is_date.append(False)
351            else:
352                try:
353                    split_key = key.split(";")
354                    if len(split_key) == 1:
355                        datetime.strptime(split_key[0], "%Y-%m-%dT%H:%M:%S")
356                        is_date.append(True)
357                        is_period.append(False)
358                    else:
359                        datetime.strptime(split_key[0], "%Y-%m-%d %H:%M:%S")
360                        datetime.strptime(split_key[1], "%Y-%m-%d %H:%M:%S")
361                        is_period.append(True)
362                        is_date.append(False)
363                except (TypeError, ValueError):
364                    is_not_period_keys.append(str(key))
365                    is_period.append(False)
366                    is_date.append(False)
367        if any(is_date):
368            if not all(is_date):
369                raise InvalidDateException(
370                    title="Invalid model",
371                    message="Temporal models should only contain date keys. "
372                    f"Invalid date(s): {','.join(is_not_period_keys)}",
373                )
374            return True
375        if any(is_period):
376            if not all(is_period):
377                raise InvalidDateException(
378                    title="Invalid model",
379                    message="Temporal models should only contain date keys. "
380                    f"Invalid date(s): {','.join(is_not_period_keys)}",
381                )
382            return True
383    return False
