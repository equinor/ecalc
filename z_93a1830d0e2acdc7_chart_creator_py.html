21 @staticmethod 

22 def from_rate_and_head_values( 

23 actual_volume_rates_m3_per_hour: list[float], 

24 heads_joule_per_kg: list[float], 

25 polytropic_efficiency: float = 1, 

26 ) -> VariableSpeedCompressorChart: 

27 """Calculate a design point such that all input data are within capacity in the corresponding generic compressor 

28 chart and where the maximum speed curve are at the "maximum" input point. 

29 

30 # Approach 

31 1. Scale the input data with an initial scaling such that for the corresponding unified generic chart maximum 

32 speed curve the maximum head is equal to maximum head in input data and maximum rate is equal to maximum rate 

33 in input data. 

34 2. Generate a variable speed compressor chart with design point (1, 1) 

35 3. Find the maximum rate of the compressor chart with design point (1, 1) using the scaled heads from the input 

36 data 

37 4. Check if any of the scaled points from the input data exceeds the capacity in the variable speed compressor 

38 chart with design point (1, 1) 

39 - If there are points below the stone wall, increase rate until no points are below stone wall. If there 

40 are any points above the maximum speed curve, iteratively increase the design rate and find the resulting 

41 design head giving no points below the stone wall. Stop when there are no points above maximum speed curve. 

42 - Else if there are points above the maximum speed curve, increase design head until one of the input points 

43 hits the stone wall. If there are no points above the maximum speed curve, reduce the head until no points 

44 are above the maximum speed curve. If there are still points above the maximum speed curve, iteratively 

45 increase the design rate and find the resulting design head giving no points below the stone wall. 

46 Stop when there are no points above maximum speed curve. 

47 5. If none of the scaled points input data exceeds the capacity in the variable speed compressor chart with 

48 design point (1, 1), there must be points that recirculate to the minimum rate/maximum head of the compressor 

49 chart, and one or more points exactly at the maximum rate/minimum head point of the maximum speed curve. In 

50 this (very unlikely) case the chart is perfect - return it. 

51 

52 Args: 

53 actual_volume_rates_m3_per_hour: volume rate values [Am3/h] 

54 heads_joule_per_kg: Head values [J/kg] 

55 polytropic_efficiency: Polytropic efficiency as a fraction between 0 and 1. 

56 

57 Returns: 

58 The resulting variable speed compressor chart 

59 """ 

60 

61 # Maximum values from unified generic chart 

62 maximum_rate_on_maximum_speed_curve_unified = UNIFIED_GENERIC_CHART_CURVE_MAXIMUM_SPEED_RATES[-1] 

63 maximum_head_on_maximum_speed_curve_unified = UNIFIED_GENERIC_CHART_CURVE_MAXIMUM_SPEED_HEADS[0] 

64 minimum_head_on_maximum_speed_curve_unified = UNIFIED_GENERIC_CHART_CURVE_MAXIMUM_SPEED_HEADS[-1] 

65 maximum_rate_on_minimum_speed_curve_unified = UNIFIED_GENERIC_CHART_CURVE_MINIMUM_SPEED_RATES[-1] 

66 maximum_change_in_rate_unified = ( 

67 maximum_rate_on_maximum_speed_curve_unified - maximum_rate_on_minimum_speed_curve_unified 

68 ) 

69 maximum_change_in_head_unified = ( 

70 maximum_head_on_maximum_speed_curve_unified - minimum_head_on_maximum_speed_curve_unified 

71 ) 

72 

73 # Maximum values from input 

74 maximum_actual_rate = np.max(actual_volume_rates_m3_per_hour) 

75 maximum_head_value = np.max(heads_joule_per_kg) 

76 

77 # Values to scale input with 

78 scaling_rate_to_unified = np.divide(maximum_actual_rate, maximum_rate_on_maximum_speed_curve_unified) 

79 scaling_head_to_unified = np.divide(maximum_head_value, maximum_head_on_maximum_speed_curve_unified) 

80 

81 initial_design_head_unified = 1 

82 initial_design_rate_unified = 1

84 def _create_compressor_chart_result_from_unified_design_point( 

85 unified_rate: float, unified_head: float 

86 ) -> CompressorChartResult:

88 design_actual_rate_m3_per_hour=unified_rate * scaling_rate_to_unified, 

89 design_head_joule_per_kg=unified_head * scaling_head_to_unified, 

90 polytropic_efficiency=polytropic_efficiency, 

91 ).evaluate_capacity_and_extrapolate_below_minimum( 

92 actual_volume_rates=actual_volume_rates_m3_per_hour, 

93 heads=heads_joule_per_kg, 

94 extrapolate_heads_below_minimum=False, 

95 )

97 initial_compressor_chart_result = _create_compressor_chart_result_from_unified_design_point( 

98 unified_rate=initial_design_rate_unified, 

99 unified_head=initial_design_head_unified, 

100 ) 

101 

102 design_rate_unified = initial_design_rate_unified 

103 design_head_unified = initial_design_head_unified 

104 

105 maximum_design_rate_unified = initial_design_rate_unified + maximum_change_in_rate_unified 

106 maximum_design_head_unified = initial_design_head_unified + maximum_change_in_head_unified 

107 

108 # Scenario 1: Point(s) below stone wall. First increase rate until there are no point(s) below stone wall. 

109 # If there are still points above maximum speed curve, iteratively increase rate slightly and 

110 # find corresponding head that gives no points below stone wall until no points are above 

111 # maximum speed curve. 

112 if initial_compressor_chart_result.any_points_below_stone_wall: 

113 # Start with high design rate (all points above stone wall), 

114 # reduce it until the first point hits the stone wall 

115 design_rate_unified = maximum_design_rate_unified - maximize_x_given_boolean_condition_function( 

116 x_min=0, 

117 x_max=maximum_design_rate_unified - design_rate_unified, 

118 bool_func=lambda x: not _create_compressor_chart_result_from_unified_design_point( 

119 unified_rate=maximum_design_rate_unified - x, 

120 unified_head=design_head_unified, 

121 ).any_points_below_stone_wall, 

122 ) 

123 while _create_compressor_chart_result_from_unified_design_point( 

124 unified_rate=design_rate_unified, 

125 unified_head=design_head_unified, 

126 ).any_points_above_maximum_speed_curve: 

127 # increase rate slightly, find head that gives no points below stone wall 

128 # repeat until no points above maximum speed curve 

129 design_rate_unified = design_rate_unified + 0.01 # increase rate with 1% of initial rate 

130 design_head_unified = maximize_x_given_boolean_condition_function( 

131 x_min=design_head_unified, 

132 x_max=maximum_design_head_unified, 

133 bool_func=lambda x: not _create_compressor_chart_result_from_unified_design_point( 

134 unified_rate=design_rate_unified, 

135 unified_head=x, 

136 ).any_points_below_stone_wall, 

137 ) 

138 

139 # Scenario 3: Point(s) above maximum speed curve. Increase head. Points can potentially end up on the stone 

140 # wall. If any points do, both rate and head must be increased according to the slope of the 

141 # stone wall (to stop points from falling under the stone wall) 

142 elif initial_compressor_chart_result.any_points_above_maximum_speed_curve: 

143 # First increase head until one point hits the stone wall 

144 design_head_unified = maximize_x_given_boolean_condition_function( 

145 x_min=design_head_unified, 

146 x_max=maximum_design_head_unified, 

147 bool_func=lambda x: not _create_compressor_chart_result_from_unified_design_point( 

148 unified_rate=design_rate_unified, 

149 unified_head=x, 

150 ).any_points_below_stone_wall, 

151 ) 

152 # If no points are above the maximum speed curve, reduce head until one point hits the maximum speed curve 

153 if not _create_compressor_chart_result_from_unified_design_point( 

154 unified_rate=design_rate_unified, 

155 unified_head=design_head_unified, 

156 ).any_points_above_maximum_speed_curve: 

157 change_in_design_head_unified = maximize_x_given_boolean_condition_function( 

158 x_min=0, 

159 x_max=maximum_change_in_head_unified, 

160 bool_func=lambda x: not _create_compressor_chart_result_from_unified_design_point( 

161 unified_rate=design_rate_unified, 

162 unified_head=design_head_unified - x, 

163 ).any_points_above_maximum_speed_curve, 

164 ) 

165 design_rate_unified = design_rate_unified - change_in_design_head_unified 

166 else: 

167 0.01 * design_rate_unified 

168 while _create_compressor_chart_result_from_unified_design_point( 

169 unified_rate=design_rate_unified, 

170 unified_head=design_head_unified, 

171 ).any_points_above_maximum_speed_curve: 

172 # increase rate slightly, find head that gives no points below stone wall 

173 # repeat until no points above maximum speed curve 

174 design_rate_unified = design_rate_unified + 0.01 # increase rate with 1% of initial rate 

175 design_head_unified = maximize_x_given_boolean_condition_function( 

176 x_min=design_head_unified, 

177 x_max=maximum_design_head_unified, 

178 bool_func=lambda x: not _create_compressor_chart_result_from_unified_design_point( 

179 unified_rate=design_rate_unified, 

180 unified_head=x, 

181 ).any_points_below_stone_wall, 

182 ) 

183 

184 design_rate, design_head = ( 

185 design_rate_unified * scaling_rate_to_unified, 

186 design_head_unified * scaling_head_to_unified, 

187 ) 

188 

189 return CompressorChartCreator.from_rate_and_head_design_point( 

190 design_actual_rate_m3_per_hour=design_rate, 

191 design_head_joule_per_kg=design_head, 

192 polytropic_efficiency=polytropic_efficiency, 

193 )

195 @staticmethod 

196 def from_rate_and_head_design_point( 

197 design_actual_rate_m3_per_hour: float, 

198 design_head_joule_per_kg: float, 

199 polytropic_efficiency: float, 

200 ) -> VariableSpeedCompressorChart: 

201 """A compressor chart based on a unified generic compressor chart scaled by a design rate and polytropic head point 

202 The chart has only two speed curves, 75% and 105% (of vendor reported maximum speed) 

203 :param design_actual_rate_m3_per_hour: Design rate [Am3/h] 

204 :param design_head_joule_per_kg: Design head [J/kg] 

205 note, head often denoted in mlc/m or kJ/kg. Input here is J/kg (not kJ/kg) as this is used for calculations 

206 :param polytropic_efficiency: Polytropic efficiency as a fraction between 0 and 1. 

207 """ 

208 logger.debug( 

209 f"Creating VariableSpeedCompressorChartGeneric with" 

210 f"design_rate: {design_actual_rate_m3_per_hour}," 

211 f"design_head: {design_head_joule_per_kg}," 

212 f"polytropic_efficiency: {polytropic_efficiency}" 

213 ) 

214 max_speed_volume_rates = design_actual_rate_m3_per_hour * UNIFIED_GENERIC_CHART_CURVE_MAXIMUM_SPEED_RATES 

215 min_speed_volume_rates = design_actual_rate_m3_per_hour * UNIFIED_GENERIC_CHART_CURVE_MINIMUM_SPEED_RATES 

216 max_speed_heads = design_head_joule_per_kg * UNIFIED_GENERIC_CHART_CURVE_MAXIMUM_SPEED_HEADS 

217 min_speed_heads = design_head_joule_per_kg * UNIFIED_GENERIC_CHART_CURVE_MINIMUM_SPEED_HEADS 

218 

219 return VariableSpeedCompressorChart( 

220 VariableSpeedChartDTO( 

221 curves=[ 

222 ChartCurveDTO( 

223 rate_actual_m3_hour=list(min_speed_volume_rates), 

224 polytropic_head_joule_per_kg=list(min_speed_heads), 

225 efficiency_fraction=[polytropic_efficiency] * len(min_speed_volume_rates), 

226 speed_rpm=75, # 75 % of max speed 

227 ), 

228 ChartCurveDTO( 

229 rate_actual_m3_hour=list(max_speed_volume_rates), 

230 polytropic_head_joule_per_kg=list(max_speed_heads), 

231 efficiency_fraction=[polytropic_efficiency] * len(max_speed_volume_rates), 

232 speed_rpm=105, # 105 % of max speed. 

233 ), 

234 ], 

235 design_head=design_head_joule_per_kg, 

236 design_rate=design_actual_rate_m3_per_hour, 

237 ) 

238 )
